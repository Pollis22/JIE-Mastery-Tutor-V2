# Agent3: Fix Railway - Use HTTP for Credentials, WebRTC for Audio

## Root Cause

We've been trying to use WebSocket for the initial handshake, but **WebRTC transport doesn't need WebSocket at all**. 

The correct flow:
1. ✅ HTTP POST → Get ephemeral credentials from OpenAI
2. ✅ WebRTC connection → Direct to OpenAI using those credentials
3. ❌ WebSocket → NOT NEEDED for WebRTC transport

---

## Solution: HTTP Endpoint for Session Creation

### Step 1: Create HTTP Session Endpoint

**File: `server/src/routes/session.ts`**

```typescript
import { Router } from 'express';

const router = Router();

// POST /api/session/realtime - Create OpenAI Realtime session
router.post('/session/realtime', async (req, res) => {
  try {
    console.log('🎬 [HTTP] Creating Realtime session');
    
    const { model = 'gpt-4o-realtime-preview', voice = 'alloy' } = req.body;

    // Validate OpenAI API key exists
    if (!process.env.OPENAI_API_KEY) {
      console.error('❌ [HTTP] Missing OPENAI_API_KEY');
      return res.status(500).json({ 
        error: 'Server configuration error',
        message: 'OpenAI API key not configured' 
      });
    }

    console.log('🔑 [HTTP] Requesting ephemeral token from OpenAI...');

    // Request ephemeral session from OpenAI
    const response = await fetch('https://api.openai.com/v1/realtime/sessions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model,
        voice,
      }),
    });

    if (!response.ok) {
      const errorText = await response.text();
      console.error('❌ [HTTP] OpenAI session creation failed:', {
        status: response.status,
        error: errorText
      });
      return res.status(response.status).json({ 
        error: 'Failed to create OpenAI session',
        details: errorText
      });
    }

    const sessionData = await response.json();

    console.log('✅ [HTTP] Session created successfully', {
      hasClientSecret: !!sessionData.client_secret?.value,
      model: sessionData.model
    });

    // Optional: Save session to database
    try {
      await saveRealtimeSession({
        userId: req.user?.id,
        studentId: req.body.studentId,
        subject: req.body.subject,
        language: req.body.language,
        ageGroup: req.body.ageGroup,
        voice,
        model,
        status: 'active'
      });
    } catch (dbError) {
      // Don't fail the request if DB save fails
      console.warn('⚠️ [HTTP] Failed to save session to DB:', dbError.message);
    }

    // Return session data with ephemeral credentials
    res.json({
      client_secret: sessionData.client_secret,
      model: sessionData.model,
      expires_at: sessionData.expires_at,
      session_id: sessionData.id
    });

  } catch (error) {
    console.error('❌ [HTTP] Session creation error:', error);
    res.status(500).json({ 
      error: 'Internal server error',
      message: error.message 
    });
  }
});
```

### Step 3: Update Client to Use HTTP + WebRTC

**File: `client/src/hooks/use-realtime-voice.tsx`**

```typescript
import { useState, useRef } from 'react';

export function useRealtimeVoice() {
  const [isConnected, setIsConnected] = useState(false);
  const [isConnecting, setIsConnecting] = useState(false);
  const [error, setError] = useState<string | null>(null);
  
  const peerConnectionRef = useRef<RTCPeerConnection | null>(null);
  const dataChannelRef = useRef<RTCDataChannel | null>(null);
  const audioElementRef = useRef<HTMLAudioElement | null>(null);

  // Connect using HTTP + WebRTC (no WebSocket needed!)
  const connect = async (config: {
    model?: string;
    voice?: string;
    studentId?: string;
    subject?: string;
    language?: string;
    ageGroup?: string;
  }) => {
    try {
      setIsConnecting(true);
      setError(null);
      console.log('🔵 Starting connection (HTTP → WebRTC)...');

      // Step 1: Get ephemeral credentials via HTTP
      console.log('🔑 Requesting session credentials...');
      const response = await fetch('/api/session/realtime', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          // Add auth header if needed:
          // 'Authorization': `Bearer ${yourAuthToken}`
        },
        body: JSON.stringify({
          model: config.model || 'gpt-4o-realtime-preview',
          voice: config.voice || 'alloy',
          studentId: config.studentId,
          subject: config.subject,
          language: config.language,
          ageGroup: config.ageGroup,
        }),
      });

      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.message || 'Failed to create session');
      }

      const { client_secret, model } = await response.json();

      if (!client_secret?.value) {
        throw new Error('No client_secret received from server');
      }

      console.log('✅ Got credentials, establishing WebRTC...');

      // Step 2: Establish WebRTC connection to OpenAI
      await connectWebRTC(client_secret.value, model);

      console.log('✅ Connected successfully!');
      setIsConnected(true);

    } catch (err) {
      console.error('❌ Connection failed:', err);
      setError(err.message);
      setIsConnected(false);
    } finally {
      setIsConnecting(false);
    }
  };

  // Establish WebRTC connection directly to OpenAI
  const connectWebRTC = async (clientSecret: string, model: string) => {
    console.log('🔵 Creating RTCPeerConnection...');

    // Create peer connection
    const pc = new RTCPeerConnection();
    peerConnectionRef.current = pc;

    // Log connection state changes
    pc.onconnectionstatechange = () => {
      console.log('🔗 Connection state:', pc.connectionState);
      if (pc.connectionState === 'failed') {
        setError('WebRTC connection failed');
        setIsConnected(false);
      }
    };

    pc.oniceconnectionstatechange = () => {
      console.log('🧊 ICE state:', pc.iceConnectionState);
    };

    // Handle incoming audio track from OpenAI
    pc.ontrack = (event) => {
      console.log('🎵 Received audio track from OpenAI');
      
      // Create or reuse audio element
      if (!audioElementRef.current) {
        const audioEl = document.createElement('audio');
        audioEl.id = 'realtime-audio';
        audioEl.autoplay = true;
        audioEl.playsInline = true;
        document.body.appendChild(audioEl);
        audioElementRef.current = audioEl;
      }

      // Attach remote audio stream
      if (audioElementRef.current.srcObject !== event.streams[0]) {
        audioElementRef.current.srcObject = event.streams[0];
      }
    };

    // Create data channel for events (optional but useful)
    const dc = pc.createDataChannel('oai-events');
    dataChannelRef.current = dc;

    dc.onopen = () => {
      console.log('✅ Data channel opened');
      
      // Optional: Send initial greeting request
      dc.send(JSON.stringify({
        type: 'response.create',
        response: {
          modalities: ['audio'],
          instructions: "Say 'Hi! I'm your tutor. What would you like to learn today?' clearly once.",
          voice: 'alloy'
        }
      }));
    };

    dc.onmessage = (event) => {
      try {
        const message = JSON.parse(event.data);
        console.log('📨 OpenAI message:', message.type);
        
        // Handle different message types
        switch (message.type) {
          case 'session.created':
          case 'session.updated':
            console.log('✅ Session ready');
            break;
          case 'error':
            console.error('❌ OpenAI error:', message.error);
            setError(message.error.message);
            break;
          case 'response.audio.delta':
            // Audio is handled via the media track, not data channel
            break;
          default:
            console.log('📬 Message:', message.type);
        }
      } catch (err) {
        console.error('❌ Failed to parse message:', err);
      }
    };

    // Add local microphone track
    console.log('🎤 Requesting microphone access...');
    const stream = await navigator.mediaDevices.getUserMedia({ 
      audio: {
        echoCancellation: true,
        noiseSuppression: true,
        autoGainControl: true,
      } 
    });

    stream.getTracks().forEach(track => {
      console.log('🎤 Adding track:', track.kind);
      pc.addTrack(track, stream);
    });

    // Create offer
    console.log('📤 Creating SDP offer...');
    const offer = await pc.createOffer();
    await pc.setLocalDescription(offer);

    // Send offer to OpenAI
    console.log('🌐 Sending offer to OpenAI...');
    const response = await fetch(
      `https://api.openai.com/v1/realtime?model=${encodeURIComponent(model)}`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${clientSecret}`,
          'Content-Type': 'application/sdp',
        },
        body: offer.sdp,
      }
    );

    if (!response.ok) {
      const errorText = await response.text();
      console.error('❌ SDP exchange failed:', response.status, errorText);
      throw new Error(`WebRTC setup failed: ${response.status}`);
    }

    // Apply remote answer
    const answerSdp = await response.text();
    await pc.setRemoteDescription({
      type: 'answer',
      sdp: answerSdp,
    });

    console.log('✅ WebRTC connection established');
  };

  // Disconnect and cleanup
  const disconnect = () => {
    console.log('🔴 Disconnecting...');

    if (dataChannelRef.current) {
      dataChannelRef.current.close();
      dataChannelRef.current = null;
    }

    if (peerConnectionRef.current) {
      peerConnectionRef.current.close();
      peerConnectionRef.current = null;
    }

    if (audioElementRef.current) {
      audioElementRef.current.srcObject = null;
      audioElementRef.current.remove();
      audioElementRef.current = null;
    }

    setIsConnected(false);
    setError(null);
  };

  return {
    connect,
    disconnect,
    isConnected,
    isConnecting,
    error,
  };
}
```

## Step 4: Update UI Component

**File: `client/src/pages/tutor-page.tsx`**

```typescript
import { useRealtimeVoice } from '../hooks/use-realtime-voice';

export function TutorPage() {
  const { connect, disconnect, isConnected, isConnecting, error } = useRealtimeVoice();

  const handleStartTutoring = async () => {
    await connect({
      model: 'gpt-4o-realtime-preview',
      voice: 'alloy',
      studentId: 'student-123',
      subject: 'Math',
      language: 'English',
      ageGroup: 'middle-school',
    });
  };

  return (
    <div>
      {!isConnected ? (
        <button 
          onClick={handleStartTutoring}
          disabled={isConnecting}
        >
          {isConnecting ? 'Connecting...' : 'Start Tutoring'}
        </button>
      ) : (
        <button onClick={disconnect}>
          End Session
        </button>
      )}

      {error && (
        <div className="error">
          Error: {error}
        </div>
      )}

      {isConnected && (
        <div className="status">
          🟢 Connected - Speak to your tutor!
        </div>
      )}
    </div>
  );
}
```

## Troubleshooting

### If HTTP endpoint returns 500:
- Check `OPENAI_API_KEY` is set in Railway
- Check Railway logs for OpenAI error details

### If WebRTC fails:
- Check browser console for CORS errors
- Verify microphone permission granted
- Check OpenAI API key has Realtime API access

### If no audio:
- Check browser autoplay policy
- Verify audio element was created
- Check `pc.ontrack` event fired
