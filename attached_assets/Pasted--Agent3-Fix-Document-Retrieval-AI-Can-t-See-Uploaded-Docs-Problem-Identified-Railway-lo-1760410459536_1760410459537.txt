```typescript
// Helper: Create chunks from text
function createChunks(text: string, chunkSize: number = 2000): string[] {
  if (!text) return [];
  
  const chunks: string[] = [];
  const words = text.split(/\s+/);
  let currentChunk = '';

  for (const word of words) {
    if ((currentChunk + ' ' + word).length > chunkSize && currentChunk) {
      chunks.push(currentChunk.trim());
      currentChunk = word;
    } else {
      currentChunk += (currentChunk ? ' ' : '') + word;
    }
  }

  if (currentChunk) {
    chunks.push(currentChunk.trim());
  }

  return chunks;
}

// Helper: Extract text from file
async function extractTextFromFile(file: Express.Multer.File): Promise<string> {
  try {
    console.log('🔍 [Extract] File type:', file.mimetype);

    // Text files
    if (file.mimetype === 'text/plain') {
      const text = file.buffer.toString('utf-8');
      console.log('✅ [Extract] Text file:', text.length, 'chars');
      return text;
    }

    // PDFs
    if (file.mimetype === 'application/pdf') {
      const pdfParse = require('pdf-parse');
      const data = await pdfParse(file.buffer);
      console.log('✅ [Extract] PDF:', data.text.length, 'chars');
      return data.text;
    }

    // Word documents
    if (file.mimetype.includes('word') || file.mimetype.includes('document')) {
      const mammoth = require('mammoth');
      const result = await mammoth.extractRawText({ buffer: file.buffer });
      console.log('✅ [Extract] Word doc:', result.value.length, 'chars');
      return result.value;
    }

    // Images (basic placeholder - add OCR later)
    if (file.mimetype.startsWith('image/')) {
      console.log('⚠️ [Extract] Image - OCR not implemented');
      return `[Image uploaded: ${file.originalname}. Please describe what help you need with this image.]`;
    }

    console.warn('⚠️ [Extract] Unsupported file type:', file.mimetype);
    return `[Document: ${file.originalname}]`;

  } catch (error) {
    console.error('❌ [Extract] Error:', error);
    return `[Document: ${file.originalname} - text extraction failed]`;
  }
}

// Helper: Save document with chunks
async function saveDocumentWithChunks(doc: any) {
  try {
    console.log('💾 [Save] Saving document:', doc.id);

    // Save main document record
    await db.query(`
      INSERT INTO documents (
        id, user_id, name, size, mime_type, 
        extracted_text, chunk_count, uploaded_at, status
      ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
    `, [
      doc.id,
      doc.userId,
      doc.name,
      doc.size,
      doc.mimeType,
      doc.extractedText,
      doc.chunkCount,
      doc.uploadedAt,
      doc.status
    ]);

    // Save chunks
    for (let i = 0; i < doc.chunks.length; i++) {
      await db.query(`
        INSERT INTO document_chunks (
          id, document_id, chunk_index, content
        ) VALUES ($1, $2, $3, $4)
      `, [
        `${doc.id}-chunk-${i}`,
        doc.id,
        i,
        doc.chunks[i]
      ]);
    }

    console.log('✅ [Save] Saved document with', doc.chunks.length, 'chunks');

  } catch (error: any) {
    if (error.code === '42P01') {
      console.warn('⚠️ [Save] Table missing, using in-memory storage');
      // Store in memory as fallback
      inMemoryDocs.set(doc.id, doc);
    } else {
      console.error('❌ [Save] Error:', error);
      throw error;
    }
  }
}

// In-memory fallback storage
const inMemoryDocs = new Map<string, any>();
```

## Fix #2: Create Database Tables

**Run this SQL in Railway Postgres:**

```sql
-- Documents table
CREATE TABLE IF NOT EXISTS documents (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL,
  name TEXT NOT NULL,
  size INTEGER,
  mime_type TEXT,
  extracted_text TEXT,
  chunk_count INTEGER DEFAULT 0,
  uploaded_at TIMESTAMPTZ DEFAULT NOW(),
  status TEXT DEFAULT 'ready'
);

CREATE INDEX IF NOT EXISTS idx_documents_user ON documents(user_id);
CREATE INDEX IF NOT EXISTS idx_documents_uploaded ON documents(uploaded_at);

-- Document chunks table (for large documents)
CREATE TABLE IF NOT EXISTS document_chunks (
  id TEXT PRIMARY KEY,
  document_id TEXT NOT NULL,
  chunk_index INTEGER NOT NULL,
  content TEXT NOT NULL,
  FOREIGN KEY (document_id) REFERENCES documents(id) ON DELETE CASCADE
);

CREATE INDEX IF NOT EXISTS idx_chunks_document ON document_chunks(document_id);
CREATE INDEX IF NOT EXISTS idx_chunks_index ON document_chunks(document_id, chunk_index);
```

Fix the function that retrieves documents for the AI:

```typescript
async function getDocumentsForAI(userId: string, documentIds: string[]) {
  console.log('🔍 [GetDocs] Fetching documents:', {
    userId,
    documentIds,
    count: documentIds.length
  });

  if (!documentIds || documentIds.length === 0) {
    console.log('⚠️ [GetDocs] No document IDs provided');
    return [];
  }

  try {
    // Try database first
    const placeholders = documentIds.map((_, i) => `$${i + 2}`).join(',');
    const query = `
      SELECT 
        d.id,
        d.name,
        d.extracted_text,
        d.chunk_count,
        array_agg(dc.content ORDER BY dc.chunk_index) as chunks
      FROM documents d
      LEFT JOIN document_chunks dc ON d.id = dc.document_id
      WHERE d.user_id = $1
      AND d.id IN (${placeholders})
      GROUP BY d.id, d.name, d.extracted_text, d.chunk_count
    `;

    console.log('📊 [GetDocs] Running query:', query);
    console.log('📊 [GetDocs] Parameters:', [userId, ...documentIds]);

    const result = await db.query(query, [userId, ...documentIds]);

    console.log('✅ [GetDocs] Found', result.rows.length, 'documents from DB');

    if (result.rows.length > 0) {
      return result.rows.map(row => ({
        id: row.id,
        name: row.name,
        content: row.extracted_text,
        chunks: row.chunks.filter(c => c !== null)
      }));
    }

    // Fallback to in-memory storage
    console.log('🔄 [GetDocs] Trying in-memory storage...');
    const docs = [];
    for (const docId of documentIds) {
      const doc = inMemoryDocs.get(docId);
      if (doc) {
        console.log('✅ [GetDocs] Found in memory:', docId);
        docs.push({
          id: doc.id,
          name: doc.name,
          content: doc.extractedText,
          chunks: doc.chunks
        });
      }
    }

    if (docs.length > 0) {
      console.log('✅ [GetDocs] Returning', docs.length, 'docs from memory');
      return docs;
    }

    console.warn('⚠️ [GetDocs] No documents found anywhere');
    return [];

  } catch (error: any) {
    console.error('❌ [GetDocs] Error:', error);
    
    if (error.code === '42P01') {
      console.warn('⚠️ [GetDocs] Tables missing, trying in-memory...');
      // Try in-memory as fallback
      const docs = [];
      for (const docId of documentIds) {
        const doc = inMemoryDocs.get(docId);
        if (doc) {
          docs.push({
            id: doc.id,
            name: doc.name,
            content: doc.extractedText,
            chunks: doc.chunks
          });
        }
      }
      return docs;
    }

    return [];
  }
}
```

## Fix #4: Include Documents in AI Instructions

**File: `server/src/routes/realtime.ts`**

When creating the OpenAI session, include document context:

```typescript
router.post('/session/realtime', async (req, res) => {
  try {
    console.log('🎬 [RealtimeAPI] Creating session');

    const {
      model = 'gpt-4o-realtime-preview',
      voice = 'alloy',
      userId,
      studentId,
      subject,
      language,
      ageGroup,
      contextDocumentIds = [] // Document IDs to include
    } = req.body;

    // Get documents if provided
    let documentContext = '';
    if (contextDocumentIds && contextDocumentIds.length > 0) {
      console.log('📚 [RealtimeAPI] Fetching documents:', contextDocumentIds);
      
      const documents = await getDocumentsForAI(userId, contextDocumentIds);
      
      console.log('📚 [RealtimeAPI] Retrieved', documents.length, 'documents');

      if (documents.length > 0) {
        documentContext = '\n\nSTUDENT UPLOADED DOCUMENTS:\n\n';
        
        for (const doc of documents) {
          documentContext += `=== ${doc.name} ===\n`;
          
          // Use chunks if available, otherwise use full content
          const chunks = doc.chunks && doc.chunks.length > 0 
            ? doc.chunks.slice(0, 3) // First 3 chunks
            : [doc.content?.substring(0, 6000) || '']; // Or first 6000 chars

          for (let i = 0; i < chunks.length; i++) {
            if (chunks[i]) {
              documentContext += `\n[Part ${i + 1}]\n${chunks[i]}\n`;
            }
          }
          
          documentContext += '\n---\n\n';
        }

        console.log('✅ [RealtimeAPI] Document context prepared:', documentContext.length, 'chars');
      } else {
        console.warn('⚠️ [RealtimeAPI] No documents found for IDs:', contextDocumentIds);
      }
    }

    // Create session with OpenAI
    const openaiResponse = await fetch('https://api.openai.com/v1/realtime/sessions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model,
        voice,
        // Include document context in instructions
        instructions: `You are a helpful AI tutor for ${ageGroup} students learning ${subject}.
${documentContext}

IMPORTANT: The student has uploaded the documents shown above. Reference them when helping!
If the student asks about homework or uploaded materials, refer to the specific content.
Be encouraging and explain concepts clearly at an appropriate level.`
      }),
    });

    if (!openaiResponse.ok) {
      const errorText = await openaiResponse.text();
      console.error('❌ [RealtimeAPI] OpenAI error:', errorText);
      return res.status(openaiResponse.status).json({ 
        error: 'OpenAI API error',
        details: errorText 
      });
    }

    const sessionData = await openaiResponse.json();

    console.log('✅ [RealtimeAPI] Session created:', {
      sessionId: sessionData.id,
      hasDocuments: documents.length > 0
    });

    // Save session to DB (fail-safe)
    try {
      await saveSession({
        id: sessionData.id,
        userId,
        studentId,
        subject,
        language,
        ageGroup,
        voice,
        model,
        documentIds: contextDocumentIds,
        documentCount: documents.length
      });
    } catch (dbError) {
      console.warn('⚠️ [RealtimeAPI] DB save failed:', dbError.message);
    }

    res.json({
      success: true,
      sessionId: sessionData.id,
      client_secret: sessionData.client_secret,
      model: sessionData.model,
      documentCount: documents.length
    });

  } catch (error: any) {
    console.error('❌ [RealtimeAPI] Error:', error);
    res.status(500).json({ 
      error: 'Session creation failed',
      message: error.message 
    });
  }
});
```
