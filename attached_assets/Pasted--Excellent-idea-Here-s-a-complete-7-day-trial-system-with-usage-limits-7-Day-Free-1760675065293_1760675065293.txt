```javascript
        }
      });
      
      stripeCustomerId = customer.id;
      
      // Save Stripe customer ID
      await pool.query(
        'UPDATE users SET stripe_customer_id = $1 WHERE id = $2',
        [stripeCustomerId, user.id]
      );
    }

    // Create checkout session with 7-day trial
    const session = await stripe.checkout.sessions.create({
      customer: stripeCustomerId,
      payment_method_types: ['card'],
      line_items: [
        {
          price: priceId, // Your Stripe price ID (e.g., price_xxxxx)
```

```javascript
// GET /subscription-success?session_id=...
router.get('/subscription-success', async (req, res) => {
  const { session_id } = req.query;

  if (!session_id) {
    return res.redirect('/pricing');
  }

  try {
    // Retrieve the session from Stripe
    const session = await stripe.checkout.sessions.retrieve(session_id);
    
    if (!session) {
      return res.redirect('/pricing');
    }

    // Get subscription details
    const subscription = await stripe.subscriptions.retrieve(session.subscription);
    
    // Calculate trial end date
    const trialEnd = new Date(subscription.trial_end * 1000); // Convert Unix timestamp
    const trialStart = new Date();

    // Update user in database
    const result = await pool.query(
      `UPDATE users SET 
        stripe_subscription_id = $1,
        subscription_status = 'trialing',
        subscription_plan = $2,
        is_trial = true,
        trial_started_at = $3,
        trial_ends_at = $4,
        trial_minutes_limit = 30,
        trial_minutes_used = 0,
        trial_reminder_sent = false,
        billing_cycle_start = $4
      WHERE stripe_customer_id = $5
      RETURNING id, email, first_name`,
      [
        subscription.id,
        subscription.items.data[0].price.nickname || 'starter',
        trialStart,
        trialEnd,
        session.customer
      ]
    );

    if (result.rows.length === 0) {
      console.error('❌ Failed to update user with trial info');
      return res.redirect('/dashboard?error=update_failed');
    }

    console.log(`✅ Trial started for user ${result.rows[0].email}`);

    // Redirect to success page
    res.redirect('/trial-started');

  } catch (error) {
    console.error('❌ Error processing subscription success:', error);
    res.redirect('/dashboard?error=processing_failed');
  }
```

## STEP 6: Track Trial Minutes Usage

```javascript
// Before starting a tutoring session, check trial limits
router.post('/session/start', async (req, res) => {
  const { userId } = req.body;

  try {
    // Get user's trial status
    const result = await pool.query(
      `SELECT 
        is_trial, 
        trial_minutes_used, 
        trial_minutes_limit,
        trial_ends_at,
        subscription_minutes_used,
        subscription_minutes_limit
      FROM users WHERE id = $1`,
      [userId]
    );

    const user = result.rows[0];

    // Check if user is on trial
    if (user.is_trial) {
      // Check if trial has expired
      if (new Date() > new Date(user.trial_ends_at)) {
        return res.status(403).json({ 
          error: 'Your trial has expired. Please subscribe to continue.',
          trialExpired: true
        });
      }

      // Check if trial minutes exhausted
      if (user.trial_minutes_used >= user.trial_minutes_limit) {
        return res.status(403).json({ 
          error: `You've used all ${user.trial_minutes_limit} trial minutes. Upgrade to continue!`,
          minutesExhausted: true,
          remainingDays: Math.ceil((new Date(user.trial_ends_at) - new Date()) / (1000 * 60 * 60 * 24))
        });
      }

      // Calculate remaining trial minutes
      const remainingMinutes = user.trial_minutes_limit - user.trial_minutes_used;
      
      res.json({
        success: true,
        isTrial: true,
        remainingMinutes: remainingMinutes,
        message: `You have ${remainingMinutes} minutes remaining in your trial`
      });

    } else {
      // Regular subscription logic
      if (user.subscription_minutes_used >= user.subscription_minutes_limit) {
        return res.status(403).json({ 
          error: 'Monthly minute limit reached. Purchase more minutes or wait for reset.',
          minutesExhausted: true
        });
      }

      res.json({
        success: true,
        isTrial: false,
        remainingMinutes: user.subscription_minutes_limit - user.subscription_minutes_used
      });
    }

  } catch (error) {
    console.error('❌ Error checking session limits:', error);
    res.status(500).json({ error: 'Failed to start session' });
  }
});

// After session ends, update minutes used
router.post('/session/end', async (req, res) => {
  const { userId, minutesUsed } = req.body;

  try {
    // Get user's trial status
    const result = await pool.query(
      'SELECT is_trial FROM users WHERE id = $1',
      [userId]
    );

    const user = result.rows[0];

    if (user.is_trial) {
      // Update trial minutes
      await pool.query(
        `UPDATE users 
         SET trial_minutes_used = trial_minutes_used + $1 
         WHERE id = $2`,
        [minutesUsed, userId]
      );
    } else {
      // Update subscription minutes
      await pool.query(
        `UPDATE users 
         SET subscription_minutes_used = subscription_minutes_used + $1 
         WHERE id = $2`,
        [minutesUsed, userId]
      );
    }

    res.json({ success: true });

  } catch (error) {
    console.error('❌ Error updating minutes:', error);
    res.status(500).json({ error: 'Failed to update session minutes' });
```

## STEP 7: Create Trial Monitoring Cron Job

Create file: `server/jobs/trial-monitor.js`

```javascript
const { Pool } = require('pg');
const { sendTrialEndingEmail } = require('../services/email');

const pool = new Pool({
  connectionString: process.env.DATABASE_URL
});

async function checkTrialReminders() {
  console.log('🔍 Checking for trials ending soon...');

  try {
    // Find users whose trial ends in 24 hours and haven't been sent a reminder
    const tomorrow = new Date(Date.now() + 24 * 60 * 60 * 1000);
    const today = new Date();

    const result = await pool.query(
      `SELECT id, email, first_name, trial_ends_at 
       FROM users 
       WHERE is_trial = true 
         AND trial_reminder_sent = false 
         AND trial_ends_at > $1 
         AND trial_ends_at <= $2`,
      [today, tomorrow]
    );

    console.log(`📧 Found ${result.rows.length} users to remind`);

    for (const user of result.rows) {
      const hoursRemaining = Math.ceil(
        (new Date(user.trial_ends_at) - new Date()) / (1000 * 60 * 60)
      );

      // Send reminder email
      const sent = await sendTrialEndingEmail(
        user.email,
        user.first_name || 'there',
        hoursRemaining
      );

      if (sent) {
        // Mark reminder as sent
        await pool.query(
          'UPDATE users SET trial_reminder_sent = true WHERE id = $1',
          [user.id]
        );
        console.log(`✅ Reminder sent to ${user.email}`);
      }
    }

  } catch (error) {
    console.error('❌ Error checking trial reminders:', error);
  }
}

// Run every hour
setInterval(checkTrialReminders, 60 * 60 * 1000);

// Run immediately on startup
checkTrialReminders();

module.exports = { checkTrialReminders };
```

```javascript
const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY);
const { sendTrialExpiredEmail } = require('../services/email');

router.post('/webhook', express.raw({ type: 'application/json' }), async (req, res) => {
  const sig = req.headers['stripe-signature'];
  let event;

  try {
    event = stripe.webhooks.constructEvent(
      req.body,
      sig,
      process.env.STRIPE_WEBHOOK_SECRET
    );
  } catch (err) {
    console.error('⚠️ Webhook signature verification failed:', err.message);
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }

  // Handle the event
  switch (event.type) {
    case 'customer.subscription.trial_will_end':
      // Stripe sends this 3 days before trial ends (backup to our own reminder)
      console.log('⏰ Trial ending soon event received');
      break;

    case 'customer.subscription.updated':
      // This fires when trial ends and becomes active subscription
      const subscription = event.data.object;
      
      if (subscription.status === 'active' && subscription.trial_end) {
        // Trial just ended and subscription is now active
        await handleTrialEnded(subscription);
      }
      break;

    case 'invoice.payment_succeeded':
      // First payment after trial
      console.log('💰 Payment succeeded:', event.data.object.id);
      break;

    case 'invoice.payment_failed':
      // Payment failed after trial
      console.log('❌ Payment failed:', event.data.object.id);
      break;

    default:
      console.log(`Unhandled event type ${event.type}`);
  }

  res.json({ received: true });
});

async function handleTrialEnded(subscription) {
  console.log('🎉 Trial ended, converting to paid subscription');

  try {
    // Update user in database
    const result = await pool.query(
      `UPDATE users SET 
        is_trial = false,
        subscription_status = 'active',
        trial_reminder_sent = true
      WHERE stripe_subscription_id = $1
      RETURNING email, first_name`,
      [subscription.id]
    );

    if (result.rows.length > 0) {
      const user = result.rows[0];
      
      // Send welcome-to-paid email
      await sendTrialExpiredEmail(user.email, user.first_name || 'there');
      
      console.log(`✅ User ${user.email} converted from trial to paid`);
    }

  } catch (error) {
    console.error('❌ Error handling trial end:', error);
  }
}

module.exports = router;
```
