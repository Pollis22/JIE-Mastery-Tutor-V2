## Fix #1: Add Document Processing Status

### Update: Show Processing State

**File: `client/src/components/document-status.tsx`**

Create a new component for document processing feedback:

```typescript
import { useState, useEffect } from 'react';

interface DocumentStatusProps {
  documents: { id: string; name: string; status?: string }[];
  isProcessing: boolean;
}

export function DocumentStatus({ documents, isProcessing }: DocumentStatusProps) {
  const [processingMessage, setProcessingMessage] = useState('');
  const [dots, setDots] = useState('');

  // Animate dots while processing
  useEffect(() => {
    if (!isProcessing) return;

    const messages = [
      'Reading your documents',
      'Understanding the content',
      'Preparing to help',
    ];

    let messageIndex = 0;
    let dotCount = 0;

    const messageInterval = setInterval(() => {
      setProcessingMessage(messages[messageIndex]);
      messageIndex = (messageIndex + 1) % messages.length;
    }, 2000);

    const dotInterval = setInterval(() => {
      dotCount = (dotCount + 1) % 4;
      setDots('.'.repeat(dotCount));
    }, 500);

    return () => {
      clearInterval(messageInterval);
      clearInterval(dotInterval);
    };
  }, [isProcessing]);

  if (!documents || documents.length === 0) return null;

  return (
    <div className="document-status">
      {isProcessing ? (
        <div className="processing-state">
          <div className="spinner">⏳</div>
          <div className="processing-text">
            <strong>{processingMessage}{dots}</strong>
            <p className="hint">This may take a few seconds...</p>
          </div>
        </div>
      ) : (
        <div className="ready-state">
          <span className="checkmark">✅</span>
          <span className="ready-text">
            {documents.length} document{documents.length > 1 ? 's' : ''} ready
          </span>
        </div>
      )}
    </div>
  );
}
```

```typescript
export function useRealtimeVoice() {
  const [isConnected, setIsConnected] = useState(false);
  const [isConnecting, setIsConnecting] = useState(false);
  const [isProcessingDocuments, setIsProcessingDocuments] = useState(false); // NEW
  const [error, setError] = useState<string | null>(null);

  const connect = useCallback(async (config) => {
    try {
      setIsConnecting(true);
      setError(null);

      // If documents provided, show processing state
      if (config.documentIds && config.documentIds.length > 0) {
        setIsProcessingDocuments(true);
        console.log('📚 Processing', config.documentIds.length, 'documents...');
      }

      // Get credentials
      const response = await fetch('/api/session/realtime', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          // ... config
          contextDocumentIds: config.documentIds || []
        }),
      });

      const data = await response.json();

      // Documents processed!
      setIsProcessingDocuments(false);
      
      if (data.documentCount > 0) {
        console.log('✅ Documents ready:', data.documentCount);
      }

      // Establish WebRTC
      await connectWebRTC(data.client_secret.value, data.model);

      setIsConnected(true);

    } catch (err) {
      setError(err.message);
      setIsProcessingDocuments(false);
    } finally {
      setIsConnecting(false);
    }
  }, []);

  return {
    connect,
    disconnect,
    isConnected,
    isConnecting,
    isProcessingDocuments, // NEW
    error,
  };
}
```

## Fix #2: Capture AI Tutor Responses in Transcript

### Critical Issue
The transcript is only showing user messages because we're not capturing AI responses from the data channel.

**File: `client/src/hooks/use-realtime-voice.tsx`**

Update the data channel handler to capture ALL messages:

```typescript
const connectWebRTC = async (clientSecret: string, model: string) => {
  // ... peer connection setup

  // Create data channel
  const dc = pc.createDataChannel('oai-events');
  dataChannelRef.current = dc;

  // Store transcript messages
  const transcriptMessages: TranscriptMessage[] = [];

  dc.onopen = () => {
    console.log('✅ [DataChannel] Opened');
    
    // Send initial greeting request
    dc.send(JSON.stringify({
      type: 'response.create',
      response: {
        modalities: ['audio'],
        instructions: "Say 'Hi! I'm your tutor. What would you like to learn today?' clearly once.",
        voice: 'alloy'
      }
    }));
  };

  dc.onmessage = (event) => {
    try {
      const message = JSON.parse(event.data);
      console.log('📨 [DataChannel] Message:', message.type);

      // Handle different message types
      switch (message.type) {
        // ========================================
        // CRITICAL: Capture user's speech
        // ========================================
        case 'conversation.item.input_audio_transcription.completed': {
          const userMessage = {
            id: message.item_id || `user-${Date.now()}`,
            role: 'user' as const,
            content: message.transcript,
            timestamp: new Date().toISOString()
          };

          console.log('👤 [Transcript] User said:', message.transcript);
          transcriptMessages.push(userMessage);

          // Update UI
          if (onTranscriptUpdate) {
            onTranscriptUpdate(userMessage);
          }

          // Save to server
          saveTranscriptMessage(sessionIdRef.current, userMessage);
          break;
        }

        // ========================================
        // CRITICAL: Capture AI tutor's responses
        // ========================================
        case 'response.audio_transcript.done':
        case 'conversation.item.created': {
          // Check if this is an assistant message
          if (message.item?.role === 'assistant' || message.type === 'response.audio_transcript.done') {
            const content = message.transcript || message.item?.content?.[0]?.transcript || '';
            
            if (content) {
              const assistantMessage = {
                id: message.item_id || message.response_id || `assistant-${Date.now()}`,
                role: 'assistant' as const,
                content: content,
                timestamp: new Date().toISOString()
              };

              console.log('🤖 [Transcript] AI said:', content);
              transcriptMessages.push(assistantMessage);

              // Update UI
              if (onTranscriptUpdate) {
                onTranscriptUpdate(assistantMessage);
              }

              // Save to server
              saveTranscriptMessage(sessionIdRef.current, assistantMessage);
            }
          }
          break;
        }

        // ========================================
        // ALTERNATIVE: Capture from audio deltas
        // ========================================
        case 'response.audio_transcript.delta': {
          // This fires for each chunk of AI speech
          // We'll accumulate and save when done
          if (!currentAssistantMessage.current) {
            currentAssistantMessage.current = {
              id: message.response_id || `assistant-${Date.now()}`,
              role: 'assistant',
              content: '',
              timestamp: new Date().toISOString()
            };
          }

          currentAssistantMessage.current.content += message.delta || '';
          
          // Update UI with partial message
          if (onTranscriptPartial) {
            onTranscriptPartial(currentAssistantMessage.current);
          }
          break;
        }

        case 'response.done': {
          // AI finished speaking - save complete message
          if (currentAssistantMessage.current) {
            console.log('🤖 [Transcript] AI complete:', currentAssistantMessage.current.content);
            
            transcriptMessages.push(currentAssistantMessage.current);

            // Update UI with final message
            if (onTranscriptUpdate) {
              onTranscriptUpdate(currentAssistantMessage.current);
            }

            // Save to server
            saveTranscriptMessage(sessionIdRef.current, currentAssistantMessage.current);

            currentAssistantMessage.current = null;
          }
          break;
        }

        case 'error': {
          console.error('❌ [OpenAI] Error:', message.error);
          break;
        }
      }
    } catch (err) {
      console.error('❌ [DataChannel] Parse error:', err);
    }
  };

  // ... rest of WebRTC setup
};
```

## Fix #3: Update Transcript UI Component

**File: `client/src/components/live-transcript.tsx`**

```typescript
import { useState, useEffect, useRef } from 'react';

interface TranscriptMessage {
  id: string;
  role: 'user' | 'assistant' | 'system';
  content: string;
  timestamp: string;
}

export function LiveTranscript({ sessionId }: { sessionId: string }) {
  const [messages, setMessages] = useState<TranscriptMessage[]>([]);
  const [partialMessage, setPartialMessage] = useState<TranscriptMessage | null>(null);
  const scrollRef = useRef<HTMLDivElement>(null);

  // Auto-scroll to bottom when new messages arrive
  useEffect(() => {
    scrollRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages, partialMessage]);

  // Add message to transcript
  const addMessage = (message: TranscriptMessage) => {
    setMessages(prev => {
      // Avoid duplicates
      if (prev.some(m => m.id === message.id)) {
        return prev;
      }
      return [...prev, message];
    });
    setPartialMessage(null);
  };

  // Update partial message (streaming)
  const updatePartial = (message: TranscriptMessage) => {
    setPartialMessage(message);
  };

  return (
    <div className="live-transcript">
      <div className="transcript-header">
        <h3>💬 Conversation</h3>
        <div className="live-badge">
          <span className="pulse-dot"></span>
          Live
        </div>
      </div>

      <div className="transcript-messages">
        {messages.length === 0 && !partialMessage && (
          <div className="empty-state">
            <p>Start speaking to begin the conversation...</p>
          </div>
        )}

        {messages.map((msg) => (
          <TranscriptMessage key={msg.id} message={msg} />
        ))}

        {/* Show partial/streaming message */}
        {partialMessage && (
          <TranscriptMessage 
            message={partialMessage} 
            isStreaming={true}
          />
        )}

        <div ref={scrollRef} />
      </div>
    </div>
  );
}

function TranscriptMessage({ 
  message, 
  isStreaming = false 
}: { 
  message: TranscriptMessage;
  isStreaming?: boolean;
}) {
  const isUser = message.role === 'user';

  return (
    <div className={`message ${message.role} ${isStreaming ? 'streaming' : ''}`}>
      <div className="message-avatar">
        {isUser ? '👤' : '🎓'}
      </div>
      <div className="message-content">
        <div className="message-header">
          <span className="message-role">
            {isUser ? 'You' : 'AI Tutor'}
          </span>
          <span className="message-time">
            {new Date(message.timestamp).toLocaleTimeString()}
          </span>
        </div>
        <div className="message-text">
          {message.content}
          {isStreaming && <span className="cursor">|</span>}
        </div>
      </div>
    </div>
  );
}
```

```typescript
router.post('/transcripts/:sessionId/messages', async (req, res) => {
  try {
    const { sessionId } = req.params;
    const { id, role, content, timestamp } = req.body;

    console.log('💾 [Transcript] Saving:', role, content.substring(0, 50));

    await db.query(`
      INSERT INTO transcripts (id, session_id, role, content, timestamp)
      VALUES ($1, $2, $3, $4, $5)
      ON CONFLICT (id) DO NOTHING
    `, [id, sessionId, role, content, timestamp]);

    res.json({ success: true });
  } catch (error: any) {
    console.warn('⚠️ [Transcript] Save failed:', error.message);
    // Don't fail the request - transcript is nice-to-have
    res.json({ success: false, error: error.message });
  }
});
```
