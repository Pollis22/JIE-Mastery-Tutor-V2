# FIX: Undefined Student Data in Instructions

## Problem Identified from Logs

```
[RealtimeWS] Session config: language=en, ageGroup=College/Adult ← Correct
[RealtimeWS] Built compact instructions for Robbie (6-8, General) ← Wrong grade!
[RealtimeWS] Instructions ready: 272 chars for undefined ← Name is undefined!
[RealtimeWS] OpenAI error: server_error ← Rejects invalid instructions
```

**Root Cause:** Student name and grade level are not being passed correctly to the instruction builder, resulting in undefined/incorrect values in the instructions template.

---

## Fix #1: Pass Correct Data to Instruction Builder

### Location: Where `buildSystemInstructions()` is called

**Current (Wrong):**
```typescript
// Wrong data being passed
const instructions = buildSystemInstructions(
  studentName,  // undefined
  gradeLevel,   // wrong value (6-8 instead of College/Adult)
  subject       // 'General'
);
```

**Fixed (Correct):**
```typescript
// Get data from session configuration
const { studentName, ageGroup, subject, language } = sessionConfig;

// Use correct values with fallbacks
const instructions = buildSystemInstructions(
  studentName || 'Student',           // Default if undefined
  ageGroup || 'General',              // Use ageGroup, not gradeLevel
  subject || 'General',               // Subject
  language || 'en'                    // Language
);

console.log(`✅ Building instructions with: name=${studentName}, age=${ageGroup}, subject=${subject}`);
```

Add validation to catch undefined values:

```typescript
// Before building instructions
function validateSessionData(sessionConfig) {
  const { studentName, ageGroup, subject } = sessionConfig;
  
  // Log what we received
  console.log('Session data received:', {
    studentName,
    ageGroup,
    subject
  });
  
  // Check for undefined values
  if (!studentName || studentName === 'undefined') {
    console.warn('⚠️ Student name is undefined, using default');
  }
  
  if (!ageGroup || ageGroup === 'undefined') {
    console.warn('⚠️ Age group is undefined, using default');
  }
  
  return {
    name: studentName || 'Student',
    age: ageGroup || 'General',
    subject: subject || 'General'
  };
}

// Use validated data
const validatedData = validateSessionData(sessionConfig);
const instructions = buildSystemInstructions(
  validatedData.name,
  validatedData.age,
  validatedData.subject
);
```

```typescript
function buildSystemInstructions(studentName, ageGroup, subject, language = 'en') {
  // Ensure no undefined values
  const name = studentName || 'Student';
  const age = ageGroup || 'General';
  const subj = subject || 'General';
  
  console.log(`Building instructions for: ${name} (${age}, ${subj})`);
  
  const instructions = `You are an expert tutor for ${age} level students, speaking with ${name} about ${subj}.

PERSONALITY:
- Warm, encouraging, and patient
- Use age-appropriate language for ${age} students
- Apply the Socratic method - guide students to discover answers
- Celebrate effort and progress enthusiastically

TEACHING APPROACH:
1. Break complex topics into smaller steps appropriate for ${age} level
2. Use real-world examples and analogies relevant to ${age} students
3. Check understanding frequently with simple questions
4. If ${name} seems stuck, offer hints before giving direct answers
5. Keep responses concise for voice conversation (30-60 seconds per turn)

CONVERSATION STYLE:
- Natural and conversational, not robotic or formal
- Enthusiastic and supportive tone
- Adapt difficulty based on ${name}'s responses
- Ask follow-up questions to ensure comprehension
- Use encouragement: "Great thinking!", "You're on the right track!"

IMPORTANT: You are in a real-time voice conversation with ${name}. Speak naturally as if you're sitting together in person.

Your goal: Help ${name} understand concepts deeply, not just get the right answer.`;

  // Validate before returning
  if (instructions.includes('undefined')) {
    console.error('❌ Instructions contain "undefined"!');
    throw new Error('Instructions have undefined values');
  }
  
  if (instructions.length < 200) {
    console.error('❌ Instructions too short');
    throw new Error('Instructions not built correctly');
  }
  
  console.log(`✅ Instructions validated: ${instructions.length} chars, no undefined values`);
  
  return instructions;
}
```

```typescript
app.post("/api/session/realtime/start", requireAuth, async (req, res) => {
  try {
    const { studentId, subject, gradeLevel, language } = req.body;
    const userId = req.session.userId;
    
    // Log what we received from client
    console.log('[RealtimeAPI] Session request:', {
      studentId,
      subject,
      gradeLevel,
      language
    });
    
    // Get user info
    const user = await db.query.users.findFirst({
      where: eq(users.id, userId!)
    });
    
    if (!user) {
      return res.status(404).json({ error: 'User not found' });
    }
    
    // Get student info
    let studentName = user.name;
    if (studentId) {
      const student = await db.query.students.findFirst({
        where: eq(students.id, studentId)
      });
      if (student) {
        studentName = student.name;
      }
    }
    
    console.log('[RealtimeAPI] Resolved student name:', studentName);
    
    // Select voice
    const voice = getVoiceForGrade(gradeLevel);
    
    // Save session to database with ALL data
    const dbSession = await db.insert(realtimeSessions).values({
      userId: userId!,
      studentId: studentId || null,
      subject: subject || 'General',
      gradeLevel: gradeLevel || 'General',  // Make sure this is saved!
      language: language || 'en',
      voice: voice,
      model: 'gpt-4o-realtime-preview',
      status: 'active',
      startedAt: new Date()
    }).returning();
    
    console.log('[RealtimeAPI] Session saved to DB:', {
      id: dbSession[0].id,
      gradeLevel: dbSession[0].gradeLevel,
      subject: dbSession[0].subject
    });
    
    // Return to client
    res.json({
      sessionId: dbSession[0].id,
      voice: voice,
      gradeLevel: gradeLevel  // Return so client knows what was set
    });
    
  } catch (error) {
    console.error('[RealtimeAPI] Error:', error);
    res.status(500).json({ error: error.message });
  }
});
```
