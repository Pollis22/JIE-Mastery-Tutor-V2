### B. Send Receipt After Payment

```typescript
// server/routes.ts - Add to webhook handler

import { Resend } from 'resend';
import { receiptEmailTemplate } from './emails/receipt';

const resend = new Resend(process.env.RESEND_API_KEY);

async function handleCheckoutComplete(session: Stripe.Checkout.Session) {
  console.log('💳 Processing checkout completion:', session.id);
  
  const userId = parseInt(session.metadata?.userId || '0');
  const planType = session.metadata?.planType;
  
  // ... existing subscription activation code ...
  
  // Get user details
  const user = await db.query.users.findFirst({
    where: eq(users.id, userId)
  });
  
  if (!user) return;
  
  // Determine plan details
  let credits = 0;
  let planName = '';
  
  switch (planType) {
    case 'starter':
      credits = 60;
      planName = 'Starter Plan';
      break;
    case 'standard':
      credits = 240;
      planName = 'Standard Plan';
      break;
    case 'pro':
      credits = 600;
      planName = 'Pro Plan';
      break;
    case 'topup':
      credits = 60;
      planName = 'Top-Up (60 minutes)';
      break;
  }
  
  // Get subscription details
  const subscription = await stripe.subscriptions.retrieve(session.subscription as string);
  
  // Send receipt email
  try {
    await resend.emails.send({
      from: 'JIE Mastery Tutor <hello@jiemastery.ai>',
      to: user.email,
      subject: `Payment Receipt - ${planName} - $${(session.amount_total! / 100).toFixed(2)}`,
      html: receiptEmailTemplate({
        customerName: user.name,
        customerEmail: user.email,
        date: new Date().toLocaleDateString('en-US', { 
          year: 'numeric', 
          month: 'long', 
          day: 'numeric' 
        }),
        planName: planName,
        amount: (session.amount_total! / 100).toFixed(2),
        credits: credits,
        nextBillingDate: subscription ? new Date(subscription.current_period_end * 1000).toLocaleDateString('en-US', { 
          year: 'numeric', 
          month: 'long', 
          day: 'numeric' 
        }) : undefined,
        invoiceUrl: session.invoice ? `https://dashboard.stripe.com/invoices/${session.invoice}` : undefined,
        subscriptionId: session.id
      })
    });
    
    console.log(`✅ Receipt email sent to ${user.email}`);
    
    // Log email in database
    await db.insert(emailLogs).values({
      userId: userId,
      type: 'receipt',
      to: user.email,
      subject: `Payment Receipt - ${planName}`,
      sentAt: new Date(),
      metadata: {
        orderId: session.id,
        amount: (session.amount_total! / 100).toFixed(2),
        plan: planName
      }
    });
    
  } catch (error) {
    console.error('❌ Failed to send receipt email:', error);
  }
}
```

## Fix #5: Add Email History to Admin Dashboard

### A. Create Email Logs Endpoint

```typescript
// server/routes.ts

// GET /api/admin/emails - Get all email logs
app.get("/api/admin/emails", requireAdmin, async (req, res) => {
  try {
    const { limit = 100, type, userId } = req.query;
    
    let query = db.select({
      id: emailLogs.id,
      userId: emailLogs.userId,
      userName: users.name,
      userEmail: users.email,
      type: emailLogs.type,
      to: emailLogs.to,
      subject: emailLogs.subject,
      sentAt: emailLogs.sentAt,
      metadata: emailLogs.metadata
    })
    .from(emailLogs)
    .leftJoin(users, eq(emailLogs.userId, users.id))
    .orderBy(desc(emailLogs.sentAt))
    .limit(parseInt(limit as string));
    
    if (type) {
      query = query.where(eq(emailLogs.type, type as string));
    }
    
    if (userId) {
      query = query.where(eq(emailLogs.userId, parseInt(userId as string)));
    }
    
    const emails = await query;
    
    res.json({ emails });
    
  } catch (error) {
    console.error('Error fetching email logs:', error);
    res.status(500).json({ error: "Failed to fetch email logs" });
  }
});

// GET /api/admin/emails/stats - Email statistics
app.get("/api/admin/emails/stats", requireAdmin, async (req, res) => {
  try {
    const totalEmails = await db.select({ count: sql`count(*)` })
      .from(emailLogs);
    
    const emailsByType = await db.select({
      type: emailLogs.type,
      count: sql`count(*)`
    })
    .from(emailLogs)
    .groupBy(emailLogs.type);
    
    const recentEmails = await db.select()
      .from(emailLogs)
      .orderBy(desc(emailLogs.sentAt))
      .limit(5);
    
    res.json({
      total: totalEmails[0].count,
      byType: emailsByType,
      recent: recentEmails
    });
    
  } catch (error) {
    console.error('Error fetching email stats:', error);
    res.status(500).json({ error: "Failed to fetch email stats" });
  }
});
```

```typescript
app.post("/api/admin/resend-receipt", requireAdmin, async (req, res) => {
  try {
    const { userId, subscriptionId } = req.body;
    
    // Get user and subscription details
    const user = await db.query.users.findFirst({

    // Determine plan name
    let planName = '';
    let credits = 0;
    
    switch (subscription.plan) {
      case 'starter':
        planName = 'Starter Plan';
        credits = 60;
        break;
      case 'standard':
        planName = 'Standard Plan';
        credits = 240;
        break;
      case 'pro':
        planName = 'Pro Plan';
        credits = 600;
        break;
    }
    
    // Send receipt
    await resend.emails.send({
      from: 'JIE Mastery Tutor <hello@jiemastery.ai>',
      to: user.email,
      subject: `Payment Receipt - ${planName} - $${subscription.price.toFixed(2)}`,
      html: receiptEmailTemplate({
        customerName: user.name,
        customerEmail: user.email,
        date: new Date(subscription.startDate).toLocaleDateString('en-US', { 
          year: 'numeric', 
          month: 'long', 
          day: 'numeric' 
        }),
        planName: planName,
        amount: subscription.price.toFixed(2),
        credits: credits,
        nextBillingDate: subscription.nextBillingDate ? new Date(subscription.nextBillingDate).toLocaleDateString('en-US', { 
          year: 'numeric', 
          month: 'long', 
          day: 'numeric' 
        }) : undefined,
        subscriptionId: subscription.stripeSubscriptionId || subscriptionId
      })
    });
    
    // Log email
    await db.insert(emailLogs).values({
      userId: userId,
      type: 'receipt',
      to: user.email,
      subject: `Payment Receipt - ${planName}`,
      sentAt: new Date(),
      metadata: {
        resent: true,
        resentBy: req.session.userId,
        originalSubscriptionId: subscriptionId
      }
    });
    
    res.json({ success: true, message: `Receipt sent to ${user.email}` });
    
  } catch (error) {
    console.error('Error resending receipt:', error);
    res.status(500).json({ error: "Failed to resend receipt" });
  }
});
```
