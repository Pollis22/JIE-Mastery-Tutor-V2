## Step 4: Create Realtime Session Server Endpoint

### Location: `server/routes.ts`

```typescript
import { OpenAI } from 'openai';

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

// POST /api/session/realtime/start - Create OpenAI Realtime session
app.post("/api/session/realtime/start", requireAuth, async (req, res) => {
  try {
    const { studentId, subject, gradeLevel, language } = req.body;
    const userId = req.session.userId;
    
    // Verify user has credits
    const user = await db.query.users.findFirst({
      where: eq(users.id, userId!)
    });
    
    if (!user || user.credits <= 0) {
      return res.status(403).json({ 
        error: "No voice minutes remaining",
        message: "Please purchase more minutes to continue tutoring."
      });
    }
    
    // Select appropriate voice based on grade level
    const voice = getVoiceForGrade(gradeLevel);
    
    // Create system prompt based on grade and subject
    const systemPrompt = getSystemPrompt(gradeLevel, subject, user.name);
    
    // Create OpenAI Realtime session
    const session = await openai.realtime.sessions.create({
      model: process.env.OPENAI_REALTIME_MODEL || 'gpt-4o-realtime-preview',
      voice: voice,
      instructions: systemPrompt,
      modalities: ['text', 'audio'],
      temperature: 0.8,
      max_response_output_tokens: 4096
    });
    
    console.log('✅ OpenAI Realtime session created:', session.id);
    
    // Save session to database
    const dbSession = await db.insert(realtimeSessions).values({
      userId: userId!,
      studentId: studentId || null,
      subject: subject || 'General',
      gradeLevel: gradeLevel || 'General',
      language: language || 'en',
      voice: voice,
      model: session.model,
      status: 'active',
      transcript: [],
      startedAt: new Date()
    }).returning();
    
    // Return session details to frontend
    res.json({
      sessionId: dbSession[0].id,
      openaiSessionId: session.id,
      clientSecret: session.client_secret.value,
      expiresAt: session.expires_at,
      voice: voice,
      creditsRemaining: user.credits
    });
    
  } catch (error) {
    console.error('❌ Error creating Realtime session:', error);
    res.status(500).json({ 
      error: "Failed to start voice session",
      details: error.message 
    });
  }
});

// Helper: Select voice based on grade level
function getVoiceForGrade(gradeLevel: string): string {
  const gradeMap: Record<string, string> = {
    'K-2': process.env.OPENAI_VOICE_K2 || 'alloy',
    '3-5': process.env.OPENAI_VOICE_35 || 'echo',
    '6-8': process.env.OPENAI_VOICE_68 || 'fable',
    '9-12': process.env.OPENAI_VOICE_912 || 'onyx',
    'College': process.env.OPENAI_VOICE_COLLEGE || 'nova'
  };
  
  return gradeMap[gradeLevel] || 'alloy';
}

// Helper: Generate system prompt
function getSystemPrompt(gradeLevel: string, subject: string, studentName: string): string {
  return `You are an expert ${gradeLevel} tutor for ${subject}. You are having a voice conversation with ${studentName}.

PERSONALITY:
- Warm, encouraging, and patient
- Age-appropriate language and examples for ${gradeLevel} students
- Use the Socratic method - guide students to discover answers
- Celebrate progress and effort enthusiastically

TEACHING APPROACH:
1. Break complex topics into smaller, digestible steps
2. Use real-world examples and analogies appropriate for ${gradeLevel}
3. Check understanding frequently with simple questions
4. If the student seems stuck, offer hints before giving direct answers
5. Keep responses concise (30-60 seconds per turn for voice)
6. Use encouragement: "Great thinking!", "You're on the right track!", etc.

CONVERSATION STYLE:
- Natural, conversational tone as if speaking to a student in person
- Avoid overly formal or robotic language
- Use excitement and enthusiasm in your voice
- Ask follow-up questions to ensure comprehension
- Adapt difficulty based on student responses

Remember: You're a supportive tutor who makes learning fun and accessible. Your goal is to help ${studentName} understand concepts deeply, not just get the right answer.`;
}
```

## Step 5: Frontend - Connect to Realtime Session

### Location: `client/src/pages/TutorPage.tsx`

```typescript
import { useEffect, useRef, useState } from 'react';

const TutorPage = () => {
  const [isSessionActive, setIsSessionActive] = useState(false);
  const [transcript, setTranscript] = useState<Message[]>([]);
  const [isConnecting, setIsConnecting] = useState(false);
  const pcRef = useRef<RTCPeerConnection | null>(null);
  const dataChannelRef = useRef<RTCDataChannel | null>(null);
  const audioRef = useRef<HTMLAudioElement>(null);
  
  const startRealtimeSession = async () => {
    try {
      setIsConnecting(true);
      
      // 1. Create session on backend
      const response = await fetch('/api/session/realtime/start', {
        method: 'POST',
        credentials: 'include',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          studentId: selectedStudent?.id,
          subject: selectedSubject,
          gradeLevel: selectedGrade,
          language: 'en'
        })
      });
      
      if (!response.ok) {
        const error = await response.json();
        throw new Error(error.message || 'Failed to start session');
      }
      
      const { clientSecret, openaiSessionId } = await response.json();
      
      // 2. Set up WebRTC connection
      const pc = new RTCPeerConnection();
      pcRef.current = pc;
      
      // 3. Get user's microphone
      const stream = await navigator.mediaDevices.getUserMedia({ 
        audio: true 
      });
      
      // 4. Add audio track to peer connection
      stream.getTracks().forEach(track => pc.addTrack(track, stream));
      
      // 5. Set up data channel for events
      const dataChannel = pc.createDataChannel('oai-events');
      dataChannelRef.current = dataChannel;
      
      dataChannel.addEventListener('message', (event) => {
        try {
          const message = JSON.parse(event.data);
          handleRealtimeMessage(message);
        } catch (error) {
          console.error('Failed to parse message:', error);
        }
      });
      
      // 6. Create and set local description
      const offer = await pc.createOffer();
      await pc.setLocalDescription(offer);
      
      // 7. Connect to OpenAI Realtime
      const baseUrl = 'https://api.openai.com/v1/realtime';
      const model = 'gpt-4o-realtime-preview-2024-10-01';
      
      const sdpResponse = await fetch(`${baseUrl}?model=${model}`, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${clientSecret}`,
          'Content-Type': 'application/sdp'
        },
        body: offer.sdp
      });
      
      if (!sdpResponse.ok) {
        throw new Error('Failed to connect to OpenAI Realtime');
      }
      
      const answerSDP = await sdpResponse.text();
      await pc.setRemoteDescription({
        type: 'answer',
        sdp: answerSDP
      });
      
      // 8. Handle incoming audio
      pc.ontrack = (event) => {
        if (audioRef.current) {
          audioRef.current.srcObject = event.streams[0];
          audioRef.current.play();
        }
      };
      
      setIsSessionActive(true);
      setIsConnecting(false);
      
      console.log('✅ Realtime session connected');
      
    } catch (error) {
      console.error('❌ Failed to start Realtime session:', error);
      alert(`Failed to start session: ${error.message}`);
      setIsConnecting(false);
    }
  };
  
  const handleRealtimeMessage = (message: any) => {
    console.log('Realtime message:', message);
    
    switch (message.type) {
      case 'conversation.item.created':
        if (message.item.role === 'user') {
          addToTranscript('student', message.item.content[0].transcript);
        }
        break;
        
      case 'response.audio_transcript.done':
        addToTranscript('tutor', message.transcript);
        break;
        
      case 'response.done':
        // Update usage/credits if needed
        break;
        
      case 'error':
        console.error('Realtime error:', message.error);
        break;
    }
  };
  
  const addToTranscript = (speaker: 'student' | 'tutor', text: string) => {
    setTranscript(prev => [...prev, {
      id: Date.now().toString(),
      speaker,
      text,
      timestamp: new Date()
    }]);
  };
  
  const endSession = () => {
    if (pcRef.current) {
      pcRef.current.close();
      pcRef.current = null;
    }
    
    if (dataChannelRef.current) {
      dataChannelRef.current.close();
      dataChannelRef.current = null;
    }
    
    setIsSessionActive(false);
    setTranscript([]);
  };
  
  return (
    <div className="tutor-page">
      <h1>Voice Tutoring Session</h1>
      
      {!isSessionActive && (
        <button 
          onClick={startRealtimeSession}
          disabled={isConnecting}
          className="btn-primary"
        >
          {isConnecting ? 'Connecting...' : 'Start Voice Session'}
        </button>
      )}
      
      {isSessionActive && (
        <>
          <button onClick={endSession} className="btn-danger">
            End Session
          </button>
          
          <div className="live-transcript">
            <h2>Live Transcript</h2>
            <div className="transcript-messages">
              {transcript.map(msg => (
                <div key={msg.id} className={`message ${msg.speaker}`}>
                  <strong>{msg.speaker === 'student' ? 'You' : 'Tutor'}:</strong>
                  <p>{msg.text}</p>
                  <span className="timestamp">
                    {msg.timestamp.toLocaleTimeString()}
                  </span>
                </div>
              ))}
            </div>
          </div>
        </>
      )}
      
      {/* Hidden audio element for playback */}
      <audio ref={audioRef} autoPlay />
    </div>
  );
};

export default TutorPage;
```

## Step 6: Track Usage and Deduct Credits

### Add endpoint to track session end and deduct credits:

```typescript
// POST /api/session/realtime/end - End session and deduct credits
app.post("/api/session/realtime/end", requireAuth, async (req, res) => {
  try {
    const { sessionId, durationSeconds } = req.body;
    const userId = req.session.userId;
    
    // Calculate minutes used (round up)
    const minutesUsed = Math.ceil(durationSeconds / 60);
    
    // Update session record
    await db.update(realtimeSessions)
      .set({
        status: 'ended',
        endedAt: new Date(),
        minutesUsed: minutesUsed
      })
      .where(eq(realtimeSessions.id, sessionId));
    
    // Deduct credits from user
    await db.update(users)
      .set({
        credits: sql`GREATEST(credits - ${minutesUsed}, 0)`,
        updatedAt: new Date()
      })
      .where(eq(users.id, userId!));
    
    // Get updated user
    const user = await db.query.users.findFirst({
      where: eq(users.id, userId!)
    });
    
    console.log(`✅ Session ended: ${minutesUsed} minutes used, ${user?.credits} remaining`);
    
    res.json({
      success: true,
      minutesUsed,
      creditsRemaining: user?.credits || 0
    });
    
  } catch (error) {
    console.error('❌ Error ending session:', error);
    res.status(500).json({ error: "Failed to end session" });
  }
});
```
