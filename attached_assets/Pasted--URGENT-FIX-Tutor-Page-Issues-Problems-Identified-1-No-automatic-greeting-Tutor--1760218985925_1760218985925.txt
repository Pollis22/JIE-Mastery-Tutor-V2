## Fix #3: Add Document Upload UI to Tutor Page

### Requirements
Students need to be able to:
1. Upload documents during an active tutoring session
2. See list of uploaded documents
3. Tell the tutor about specific documents to reference
4. Remove documents if needed

### Implementation

#### A. Add Upload Section to Tutor Page

Insert this section ABOVE the transcript area:

```tsx
import { Upload, File, X } from 'lucide-react';

const DocumentUploadSection = () => {
  const [uploadedDocs, setUploadedDocs] = useState<Document[]>([]);
  const [isUploading, setIsUploading] = useState(false);
  
  const handleFileUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const files = e.target.files;
    if (!files || files.length === 0) return;
    
    setIsUploading(true);
    
    try {
      const formData = new FormData();
      formData.append('file', files[0]);
      formData.append('studentId', selectedStudent.id);
      formData.append('subject', selectedSubject);
      formData.append('gradeLevel', selectedGrade);
      
      const response = await fetch('/api/documents/upload', {
        method: 'POST',
        body: formData,
        credentials: 'include'
      });
      
      if (!response.ok) throw new Error('Upload failed');
      
      const result = await response.json();
      setUploadedDocs([...uploadedDocs, result.document]);
      
      // Show success message
      alert(`✅ ${files[0].name} uploaded successfully!`);
      
      // Optionally notify the tutor
      addToTranscript({
        speaker: 'system',
        text: `📄 Document uploaded: ${files[0].name}`,
        timestamp: new Date()
      });
      
    } catch (error) {
      console.error('Upload error:', error);
      alert('❌ Failed to upload document. Please try again.');
    } finally {
      setIsUploading(false);
    }
  };
  
  const removeDocument = async (docId: string) => {
    try {
      await fetch(`/api/documents/${docId}`, {
        method: 'DELETE',
        credentials: 'include'
      });
      setUploadedDocs(uploadedDocs.filter(doc => doc.id !== docId));
    } catch (error) {
      console.error('Delete error:', error);
      alert('Failed to remove document');
    }
  };
  
  return (
    <div className="bg-white rounded-lg shadow-md p-4 mb-4">
      <div className="flex items-center justify-between mb-3">
        <h3 className="text-lg font-semibold flex items-center gap-2">
          <File className="w-5 h-5" />
          Study Materials
        </h3>
        <label className="btn btn-sm bg-blue-600 text-white cursor-pointer flex items-center gap-2">
          <Upload className="w-4 h-4" />
          Upload Document
          <input
            type="file"
            className="hidden"
            accept=".pdf,.docx,.doc,.txt,.png,.jpg,.jpeg"
            onChange={handleFileUpload}
            disabled={isUploading}
          />
        </label>
      </div>
      
      {uploadedDocs.length === 0 ? (
        <p className="text-sm text-gray-500 text-center py-4">
          No documents uploaded yet. Upload homework, notes, or study guides to get personalized help!
        </p>
      ) : (
        <div className="space-y-2">
          {uploadedDocs.map((doc) => (
            <div key={doc.id} className="flex items-center justify-between bg-gray-50 p-3 rounded">
              <div className="flex items-center gap-2">
                <File className="w-4 h-4 text-blue-600" />
                <span className="text-sm font-medium">{doc.filename}</span>
                <span className="text-xs text-gray-500">
                  {(doc.fileSize / 1024).toFixed(1)} KB
                </span>
              </div>
              <button
                onClick={() => removeDocument(doc.id)}
                className="text-red-500 hover:text-red-700"
              >
                <X className="w-4 h-4" />
              </button>
            </div>
          ))}
        </div>
      )}
      
      {isUploading && (
        <div className="text-center py-2">
          <span className="text-sm text-blue-600">Uploading...</span>
        </div>
      )}
    </div>
  );
};
```
