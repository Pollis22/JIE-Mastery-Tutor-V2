## Fix #1: Handle Invalid Stripe Customer IDs

### Location
`server/routes.ts` - Stripe checkout session creation endpoint

### Problem
The code tries to use an existing `stripeCustomerId` from the database, but that customer doesn't exist in Stripe anymore (likely from account migration).

### Solution

Update the checkout session creation logic to handle invalid customers:

```typescript
// POST /api/create-checkout-session
app.post("/api/create-checkout-session", requireAuth, async (req, res) => {
  try {
    const { priceId, planType } = req.body;
    const userId = req.session.userId;
    
    if (!userId) {
      return res.status(401).json({ error: "Not authenticated" });
    }
    
    // Get user from database
    const user = await db.query.users.findFirst({
      where: eq(users.id, userId)
    });
    
    if (!user) {
      return res.status(404).json({ error: "User not found" });
    }
    
    let customerId = user.stripeCustomerId;
    
    // CRITICAL FIX: Verify customer exists in Stripe, create new one if invalid
    if (customerId) {
      try {
        await stripe.customers.retrieve(customerId);
        console.log(`✅ Using existing Stripe customer: ${customerId}`);
      } catch (error) {
        console.warn(`⚠️ Invalid Stripe customer ID: ${customerId}. Creating new customer.`);
        customerId = null; // Reset to create new customer
      }
    }
    
    // Create new Stripe customer if needed
    if (!customerId) {
      const customer = await stripe.customers.create({
        email: user.email,
        name: user.name,
        metadata: {
          userId: userId.toString()
        }
      });
      
      customerId = customer.id;
      
      // Update database with new customer ID
      await db.update(users)
        .set({ stripeCustomerId: customerId })
        .where(eq(users.id, userId));
      
      console.log(`✅ Created new Stripe customer: ${customerId}`);
    }
    
    // Create checkout session
    const session = await stripe.checkout.sessions.create({
      customer: customerId,
      payment_method_types: ['card'],
      line_items: [
        {
          price: priceId,
          quantity: 1,
        },
      ],
      mode: planType === 'topup' ? 'payment' : 'subscription',
      success_url: `${process.env.BASE_URL || 'https://jie-mastery-tutor-v2-production.up.railway.app'}/dashboard?session_id={CHECKOUT_SESSION_ID}`,
      cancel_url: `${process.env.BASE_URL || 'https://jie-mastery-tutor-v2-production.up.railway.app'}/pricing`,
      metadata: {
        userId: userId.toString(),
        planType: planType
      }
    });
    
    res.json({ url: session.url });
    
  } catch (error) {
    console.error('Checkout session creation error:', error);
    res.status(500).json({ 
      error: 'Failed to create checkout session',
      details: error.message 
    });
  }
});
```

## Fix #2: Clean Up Invalid Customer IDs in Database

### Create Database Cleanup Script

Add a one-time migration to fix existing users with invalid Stripe customer IDs:

```typescript
// server/db/cleanup-stripe-customers.ts
import { db } from './index';
import { users } from './schema';
import { isNotNull } from 'drizzle-orm';
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2023-10-16'
});

export async function cleanupInvalidStripeCustomers() {
  console.log('🔍 Checking for invalid Stripe customer IDs...');
  
  // Get all users with a stripeCustomerId
  const usersWithStripeId = await db.query.users.findMany({
    where: isNotNull(users.stripeCustomerId)
  });
  
  let fixedCount = 0;
  
  for (const user of usersWithStripeId) {
    try {
      // Try to retrieve customer from Stripe
      await stripe.customers.retrieve(user.stripeCustomerId!);
      console.log(`✅ Valid customer: ${user.email}`);
    } catch (error) {
      // Customer doesn't exist, clear it from database
      console.log(`❌ Invalid customer for ${user.email}, clearing...`);
      
      await db.update(users)
        .set({ stripeCustomerId: null })
        .where(eq(users.id, user.id));
      
      fixedCount++;
    }
  }
  
  console.log(`✅ Cleanup complete. Fixed ${fixedCount} users.`);
}
```

### Run the cleanup (add to server startup or run manually):

## Fix #4: Add Better Error Handling on Frontend

### Update Pricing Page (`client/src/pages/Pricing.tsx`):

```typescript
const handleSubscribe = async (priceId: string, planType: string) => {
  try {
    setIsLoading(true);
    
    const response = await fetch('/api/create-checkout-session', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      credentials: 'include',
      body: JSON.stringify({ priceId, planType })
    });
    
    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.error || error.details || 'Failed to create checkout session');
    }
    
    const { url } = await response.json();
    
    if (!url) {
      throw new Error('No checkout URL received');
    }
    
    // Redirect to Stripe Checkout
    window.location.href = url;
    
  } catch (error) {
    console.error('Checkout error:', error);
    
    // Show user-friendly error message
    alert(`❌ Unable to start checkout: ${error.message}\n\nPlease try again or contact support if the issue persists.`);
    
    setIsLoading(false);
  }
};
```
