## Fix #1: Update Disconnect Function in Hook

**File: `client/src/hooks/use-realtime-voice.tsx`**

Replace the `disconnect` function with this improved version:

```typescript
const disconnect = useCallback(() => {
  console.log('🔴 [RealtimeVoice] Disconnecting...');

  try {
    // 1. Close data channel first
    if (dataChannelRef.current) {
      console.log('🔴 [DataChannel] Closing...');
      try {
        dataChannelRef.current.close();
      } catch (e) {
        console.warn('⚠️ [DataChannel] Error closing:', e);
      }
      dataChannelRef.current = null;
    }

    // 2. Close peer connection
    if (peerConnectionRef.current) {
      console.log('🔴 [WebRTC] Closing peer connection...');
      try {
        peerConnectionRef.current.close();
      } catch (e) {
        console.warn('⚠️ [WebRTC] Error closing:', e);
      }
      peerConnectionRef.current = null;
    }

    // 3. Stop microphone tracks
    if (micStreamRef.current) {
      console.log('🔴 [Microphone] Stopping tracks...');
      micStreamRef.current.getTracks().forEach(track => {
        try {
          track.stop();
          console.log('✅ [Microphone] Track stopped');
        } catch (e) {
          console.warn('⚠️ [Microphone] Error stopping track:', e);
        }
      });
      micStreamRef.current = null;
    }

    // 4. Remove and cleanup audio element
    if (audioElementRef.current) {
      console.log('🔴 [Audio] Removing element...');
      try {
        audioElementRef.current.pause();
        audioElementRef.current.srcObject = null;
        audioElementRef.current.remove();
      } catch (e) {
        console.warn('⚠️ [Audio] Error removing:', e);
      }
      audioElementRef.current = null;
    }

    console.log('✅ [RealtimeVoice] Disconnected successfully');
    
    // Update state
    setIsConnected(false);
    setError(null);

  } catch (error) {
    console.error('❌ [RealtimeVoice] Error during disconnect:', error);
    // Force state update even if cleanup failed
    setIsConnected(false);
  }
}, []);
```

## Fix #4: Update UI to Show Proper State

**File: `client/src/pages/tutor-page.tsx`**

Make sure the UI properly reflects the connection state:

```typescript
export function TutorPage() {
  const { connect, disconnect, isConnected, isConnecting, error } = useRealtimeVoice();
  const [sessionTime, setSessionTime] = useState(0);
  
  // Timer for session duration
  useEffect(() => {
    if (!isConnected) return;
    
    const interval = setInterval(() => {
      setSessionTime(prev => prev + 1);
    }, 1000);
    
    return () => {
      clearInterval(interval);
      setSessionTime(0);
    };
  }, [isConnected]);

  const handleEndSession = () => {
    if (window.confirm('Are you sure you want to end this tutoring session?')) {
      disconnect();
    }
  };

  const formatTime = (seconds: number) => {
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins}:${secs.toString().padStart(2, '0')}`;
  };

  return (
    <div>
      {isConnected ? (
        <div className="session-active">
          <div className="status">
            <span className="indicator">🟢</span>
            <span>Session Active</span>
            <span className="time">{formatTime(sessionTime)}</span>
          </div>
          
          <button 
            onClick={handleEndSession}
            className="btn-danger"
          >
            End Session
          </button>
        </div>
      ) : (
        <div className="session-setup">
          {/* Your form fields */}
          <button 
            onClick={handleStart}
            disabled={isConnecting}
            className="btn-primary"
          >
            {isConnecting ? 'Connecting...' : 'Start Tutoring Session'}
          </button>
        </div>
      )}

      {error && (
        <div className="error-banner">
          ❌ {error}
          <button onClick={() => disconnect()}>Dismiss</button>
        </div>
      )}
    </div>
  );
}
```

```typescript
export function useRealtimeVoice() {
  // ... existing code

  // Cleanup on unmount or page unload
  useEffect(() => {
    const handleBeforeUnload = () => {
      disconnect();
    };

    window.addEventListener('beforeunload', handleBeforeUnload);
    
    return () => {
      window.removeEventListener('beforeunload', handleBeforeUnload);
      disconnect(); // Cleanup on component unmount
    };
  }, [disconnect]);

  // ... rest of hook
}
```

## Common Issues & Solutions

### Issue 1: Microphone stays active after disconnect

**Problem:** Green dot in Chrome tab persists

**Solution:** Make sure `track.stop()` is called:
```typescript
stream.getTracks().forEach(track => {
  console.log('Stopping track:', track.kind, track.label);
  track.stop();
});
```

### Issue 2: Audio element not removed

**Problem:** `<audio>` element stays in DOM

**Solution:** 
```typescript
if (audioElementRef.current) {
  audioElementRef.current.pause();
  audioElementRef.current.srcObject = null;
  audioElementRef.current.remove();
  audioElementRef.current = null;
}
```

### Issue 3: State not updating

**Problem:** UI still shows "connected" after disconnect

**Solution:** Ensure `setIsConnected(false)` is called in `finally` block:
```typescript
try {
  // cleanup
} catch (error) {
  console.error(error);
} finally {
  setIsConnected(false); // Always update state
}
```

## Quick Reference

**Disconnect checklist:**
1. ✅ Close data channel
2. ✅ Close peer connection
3. ✅ Stop microphone tracks
4. ✅ Remove audio element
5. ✅ Update state
6. ✅ Clear any timers
7. ✅ Remove event listeners

**Do NOT:**
- ❌ Call non-existent API endpoints
- ❌ Wait for server response before cleanup
- ❌ Leave microphone running
- ❌ Leave audio element in DOM
