## Fix #3: Validate Each Parameter

Add validation before sending:

```typescript
function validateSessionConfig(config) {
  const session = config.session;
  
  // Check required fields
  const required = ['modalities', 'instructions', 'voice'];
  for (const field of required) {
    if (!session[field]) {
      throw new Error(`Missing required field: ${field}`);
    }
  }
  
  // Check instructions length
  if (session.instructions.length < 400) {
    throw new Error(`Instructions too short: ${session.instructions.length} chars`);
  }
  
  // Check modalities
  if (!Array.isArray(session.modalities) || session.modalities.length === 0) {
    throw new Error('Modalities must be non-empty array');
  }
  
  // Check voice
  const validVoices = ['alloy', 'echo', 'fable', 'onyx', 'nova', 'shimmer'];
  if (!validVoices.includes(session.voice)) {
    throw new Error(`Invalid voice: ${session.voice}`);
  }
  
  // Check audio formats
  if (session.input_audio_format && session.input_audio_format !== 'pcm16') {
    console.warn('⚠️ Non-standard input audio format');
  }
  
  console.log('✅ Session config validated');
  return true;
}

// Use before sending
try {
  validateSessionConfig(sessionConfig);
  ws.send(JSON.stringify(sessionConfig));
} catch (error) {
  console.error('❌ Invalid config:', error);
  throw error;
}
```

```typescript
ws.on('message', (data) => {
  const message = JSON.parse(data.toString());
  
  switch (message.type) {
    case 'error':
      console.error('[RealtimeWS] OpenAI Error Details:', {
        type: message.error.type,
        code: message.error.code,
        message: message.error.message,
        param: message.error.param,
        event_id: message.event_id
      });
      
      // Check if it's a retryable error
      if (message.error.type === 'server_error') {
        console.log('⚠️ OpenAI internal error - may be temporary');
        console.log('Session ID for support:', message.error.message.match(/sess_[a-zA-Z0-9]+/)?.[0]);
      }
      
      // Try to identify what caused the error
      if (message.error.param) {
        console.error(`❌ Problem with parameter: ${message.error.param}`);
      }
      
      break;
      
    case 'session.created':
      console.log('[RealtimeWS] ✅ Session created:', message.session);
      // Log what OpenAI thinks the session config is
      console.log('OpenAI session config:', {
        model: message.session.model,
        voice: message.session.voice,
        modalities: message.session.modalities
      });
      break;
      
    case 'session.updated':
      console.log('[RealtimeWS] ✅ Session updated');
      break;
  }
```

## Fix #7: Add Retry Logic

OpenAI says "We recommend you retry your request" - so add retry:

```typescript
async function connectWithRetry(sessionId, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      console.log(`[RealtimeWS] Connection attempt ${attempt}/${maxRetries}`);
      
      const ws = await createWebSocketConnection(sessionId);
      
      // Wait for session.updated confirmation
      const updated = await waitForSessionUpdate(ws, 5000); // 5 second timeout
      
      if (updated) {
        console.log('✅ Session connected successfully');
        return ws;
      }
      
    } catch (error) {
      console.error(`❌ Attempt ${attempt} failed:`, error);
      
      if (attempt < maxRetries) {
        const delay = attempt * 1000; // Exponential backoff
        console.log(`Retrying in ${delay}ms...`);
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
  }
  
  throw new Error('Failed to connect after max retries');
}
```
