## Feature #1: Live Streaming Transcripts

### What It Does
Shows words appearing in real-time as the AI speaks (like YouTube captions), giving students immediate visual feedback.

### Implementation

**File: `client/src/components/live-transcript.tsx`**

```typescript
import { useState, useEffect, useRef } from 'react';

interface TranscriptMessage {
  id: string;
  role: 'user' | 'assistant' | 'system';
  content: string;
  timestamp: string;
  isPartial?: boolean;
}

export function LiveTranscript({ sessionId }: { sessionId: string }) {
  const [messages, setMessages] = useState<TranscriptMessage[]>([]);
  const [currentPartial, setCurrentPartial] = useState<string>('');
  const scrollRef = useRef<HTMLDivElement>(null);

  // Auto-scroll to bottom
  useEffect(() => {
    scrollRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages, currentPartial]);

  // Handle incoming messages
  const addMessage = (message: TranscriptMessage) => {
    setMessages(prev => [...prev, message]);
    setCurrentPartial('');
  };

  // Handle partial (streaming) text
  const updatePartial = (text: string, role: 'user' | 'assistant') => {
    setCurrentPartial(text);
  };

  return (
    <div className="live-transcript">
      <div className="transcript-header">
        <h3>📝 Conversation</h3>
        <div className="live-indicator">
          <span className="pulse"></span>
          <span>Live</span>
        </div>
      </div>

      <div className="transcript-messages">
        {messages.map((msg) => (
          <div 
            key={msg.id} 
            className={`message message-${msg.role}`}
          >
            <div className="message-avatar">
              {msg.role === 'user' ? '👤' : '🤖'}
            </div>
            <div className="message-content">
              <div className="message-header">
                <span className="message-role">
                  {msg.role === 'user' ? 'You' : 'AI Tutor'}
                </span>
                <span className="message-time">
                  {new Date(msg.timestamp).toLocaleTimeString()}
                </span>
              </div>
              <div className="message-text">
                <RichContent content={msg.content} />
              </div>
            </div>
          </div>
        ))}

        {/* Streaming partial message */}
        {currentPartial && (
          <div className="message message-assistant streaming">
            <div className="message-avatar">🤖</div>
            <div className="message-content">
              <div className="message-header">
                <span className="message-role">AI Tutor</span>
                <span className="typing-indicator">typing...</span>
              </div>
              <div className="message-text">
                {currentPartial}
                <span className="cursor">|</span>
              </div>
            </div>
          </div>
        )}

        <div ref={scrollRef} />
      </div>

      {/* Export button */}
      <div className="transcript-actions">
        <button onClick={() => exportTranscript(sessionId)}>
          📥 Export Transcript
        </button>
      </div>
    </div>
  );
}

// Rich content renderer (for Feature #4)
function RichContent({ content }: { content: string }) {
  // Detect and render math equations
  if (content.includes('$$') || content.includes('\\(')) {
    return <MathRenderer content={content} />;
  }

  // Detect and render code blocks
  if (content.includes('```')) {
    return <CodeRenderer content={content} />;
  }

  // Regular text with markdown
  return <MarkdownRenderer content={content} />;
}

async function exportTranscript(sessionId: string) {
  const response = await fetch(`/api/transcripts/${sessionId}/export`);
  const blob = await response.blob();
  const url = window.URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `transcript-${sessionId}.pdf`;
  a.click();
}

## Feature #2: Smart Document Highlighting

### What It Does
When AI references a document, highlights the relevant section in the document viewer.

**File: `client/src/components/document-viewer.tsx`**

```typescript
import { useState, useEffect } from 'react';

interface DocumentReference {
  documentId: string;
  page?: number;
  section?: string;
  timestamp: string;
}

export function DocumentViewer({ 
  documents, 
  currentReference 
}: { 
  documents: Document[];
  currentReference?: DocumentReference;
}) {
  const [selectedDoc, setSelectedDoc] = useState<Document | null>(null);
  const [highlightedSection, setHighlightedSection] = useState<string | null>(null);

  // Auto-switch to referenced document
  useEffect(() => {
    if (currentReference) {
      const doc = documents.find(d => d.id === currentReference.documentId);
      if (doc) {
        setSelectedDoc(doc);
        setHighlightedSection(currentReference.section || null);
        
        // Auto-hide highlight after 5 seconds
        setTimeout(() => setHighlightedSection(null), 5000);
      }
    }
  }, [currentReference, documents]);

  return (
    <div className="document-viewer">
      {/* Document tabs */}
      <div className="document-tabs">
        {documents.map(doc => (
          <button
            key={doc.id}
            className={`tab ${selectedDoc?.id === doc.id ? 'active' : ''}`}
            onClick={() => setSelectedDoc(doc)}
          >
            📄 {doc.name}
            {currentReference?.documentId === doc.id && (
              <span className="reference-indicator">👁️</span>
            )}
          </button>
        ))}
      </div>

      {/* Document content */}
      {selectedDoc && (
        <div className="document-content">
          {currentReference?.page && (
            <div className="reference-banner">
              <span>👁️ AI is looking at page {currentReference.page}</span>
            </div>
          )}
          
          <iframe
            src={`/api/documents/${selectedDoc.id}/view`}
            className="document-frame"
          />
          
          {highlightedSection && (
            <div className="highlight-overlay">
              <div className="highlight-box">
                {highlightedSection}
              </div>
            </div>
          )}
        </div>
      )}
    </div>
  );
}

// Parse AI responses to detect document references
function parseDocumentReferences(text: string, documents: Document[]) {
  const references: DocumentReference[] = [];
  
  // Detect patterns like "Looking at problem 3" or "On page 2"
  const pageMatch = text.match(/page (\d+)/i);
  const problemMatch = text.match(/problem (\d+)/i);
  const sectionMatch = text.match(/section (\d+)/i);
  
  if (pageMatch || problemMatch || sectionMatch) {
    references.push({
      documentId: documents[0]?.id, // Assume first doc
      page: pageMatch ? parseInt(pageMatch[1]) : undefined,
      section: problemMatch ? `Problem ${problemMatch[1]}` : 
               sectionMatch ? `Section ${sectionMatch[1]}` : undefined,
      timestamp: new Date().toISOString()
    });
  }
  
  return references;
}

import { Router } from 'express';

const router = Router();

router.get('/insights/:userId', async (req, res) => {
  try {
    const { userId } = req.params;

    // Get session stats
    const sessions = await db.query(`
      SELECT 
        COUNT(*) as total_sessions,
        SUM(minutes_used) as total_minutes,
        AVG(minutes_used) as avg_session_length,
        MAX(started_at) as most_recent
      FROM realtime_sessions
      WHERE user_id = $1
    `, [userId]);

    // Get topics discussed (extract from transcripts)
    const topics = await db.query(`
      SELECT 
        subject,
        COUNT(*) as count
      FROM realtime_sessions
      WHERE user_id = $1
      GROUP BY subject
      ORDER BY count DESC
      LIMIT 10
    `, [userId]);

    // Count documents
    const documents = await db.query(`
      SELECT COUNT(*) as count
      FROM documents
      WHERE user_id = $1
    `, [userId]);

    // Calculate streak
    const streak = await calculateStreak(userId);

    res.json({
      totalSessions: parseInt(sessions.rows[0].total_sessions),
      totalMinutes: parseFloat(sessions.rows[0].total_minutes) || 0,
      averageSessionLength: parseFloat(sessions.rows[0].avg_session_length) || 0,
      mostRecentSession: sessions.rows[0].most_recent,
      topicsDiscussed: topics.rows,
      documentsUploaded: parseInt(documents.rows[0].count),
      streak,
    });

  } catch (error) {
    console.error('Insights error:', error);
    res.status(500).json({ error: error.message });
  }
});

async function calculateStreak(userId: string) {
  // Get all session dates
  const result = await db.query(`
    SELECT DATE(started_at) as session_date
    FROM realtime_sessions
    WHERE user_id = $1
    ORDER BY session_date DESC
  `, [userId]);

  if (result.rows.length === 0) return 0;

  let streak = 0;
  let currentDate = new Date();
  currentDate.setHours(0, 0, 0, 0);

  for (const row of result.rows) {
    const sessionDate = new Date(row.session_date);
    const diffDays = Math.floor(
      (currentDate.getTime() - sessionDate.getTime()) / (1000 * 60 * 60 * 24)
    );

    if (diffDays === streak) {
      streak++;
    } else {
      break;
    }
  }

  return streak;
}

export default router;

## Feature #4: Rich Content Rendering

### What It Does
Properly displays math equations, code blocks, and diagrams in transcripts.

**File: `client/src/components/math-renderer.tsx`**

```typescript
import { useEffect, useRef } from 'react';
import katex from 'katex';
import 'katex/dist/katex.min.css';

export function MathRenderer({ content }: { content: string }) {
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!containerRef.current) return;

    try {
      // Replace inline math: \(...\) or $...$
      let html = content.replace(
        /\\\((.*?)\\\)|\$([^$]+)\$/g,
        (match, inline1, inline2) => {
          const math = inline1 || inline2;
          return katex.renderToString(math, { displayMode: false });
        }
      );

      // Replace display math: \[...\] or $$...$$
      html = html.replace(
        /\\\[(.*?)\\\]|\$\$(.*?)\$\$/gs,
        (match, display1, display2) => {
          const math = display1 || display2;
          return katex.renderToString(math, { displayMode: true });
        }
      );

      containerRef.current.innerHTML = html;
    } catch (error) {
      console.error('Math rendering error:', error);
      containerRef.current.innerHTML = content;
    }
  }, [content]);

  return <div ref={containerRef} className="math-content" />;
}

```typescript
import Prism from 'prismjs';
import 'prismjs/themes/prism-tomorrow.css';

export function CodeRenderer({ content }: { content: string }) {
  // Extract code blocks
  const blocks = content.split('```');
  
  return (
    <div className="code-content">
      {blocks.map((block, i) => {
        if (i % 2 === 0) {
          // Regular text
          return <p key={i}>{block}</p>;
        } else {
          // Code block
          const [lang, ...codeLines] = block.split('\n');
          const code = codeLines.join('\n').trim();
          const highlighted = Prism.highlight(
            code,
            Prism.languages[lang] || Prism.languages.javascript,
            lang
          );

          return (
            <div key={i} className="code-block">
              <div className="code-header">
                <span className="language">{lang}</span>
                <button onClick={() => navigator.clipboard.writeText(code)}>
                  📋 Copy
                </button>
              </div>
              <pre>
                <code dangerouslySetInnerHTML={{ __html: highlighted }} />
              </pre>
            </div>
          );
        }
      })}
    </div>
  );
}

## Feature #5: Export & Share

### What It Does
Download transcripts as PDF, share progress with parents/teachers.

**File: `server/src/routes/export.ts`**

```typescript
import { Router } from 'express';
import PDFDocument from 'pdfkit';

const router = Router();

router.get('/transcripts/:sessionId/export', async (req, res) => {
  try {
    const { sessionId } = req.params;

    // Get session details
    const session = await db.query(`
      SELECT * FROM realtime_sessions WHERE id = $1
    `, [sessionId]);

    // Get transcript messages
    const messages = await db.query(`
      SELECT * FROM transcripts 
      WHERE session_id = $1 
      ORDER BY timestamp ASC
    `, [sessionId]);

    // Create PDF
    const doc = new PDFDocument();
    res.setHeader('Content-Type', 'application/pdf');
    res.setHeader(
      'Content-Disposition',
      `attachment; filename=transcript-${sessionId}.pdf`
    );

    doc.pipe(res);

    // Add header
    doc.fontSize(20).text('JIE Mastery Tutor - Session Transcript', {
      align: 'center'
    });
    doc.moveDown();

    // Add session info
    doc.fontSize(12);
    doc.text(`Student: ${session.rows[0].student_id}`);
    doc.text(`Subject: ${session.rows[0].subject}`);
    doc.text(`Date: ${new Date(session.rows[0].started_at).toLocaleString()}`);
    doc.text(`Duration: ${session.rows[0].minutes_used} minutes`);
    doc.moveDown(2);

    // Add conversation
    doc.fontSize(10);
    messages.rows.forEach(msg => {
      const time = new Date(msg.timestamp).toLocaleTimeString();
      const role = msg.role === 'user' ? 'Student' : 'AI Tutor';
      
      doc.font('Helvetica-Bold').text(`[${time}] ${role}:`);
      doc.font('Helvetica').text(msg.content);
      doc.moveDown();
    });

    doc.end();

  } catch (error) {
    console.error('Export error:', error);
    res.status(500).json({ error: error.message });
  }
});

export default router;
```
