# Agent3 Task: Implement Learning Sessions Storage & Transcript Access

## Goal
Store all voice tutoring sessions with full transcripts and make them accessible to users for review and studying.

## User Requirements
- Dashboard shows **last 10 sessions** (quick overview)
- Dedicated **"Learning Sessions" page** shows all sessions from last 30 days
- Full transcripts available for each session
- Searchable by student, subject, date
- Auto-delete sessions older than 30 days

---

```sql
  ON realtime_sessions(student_id, started_at DESC);

CREATE INDEX IF NOT EXISTS idx_sessions_status_started 
  ON realtime_sessions(status, started_at DESC);

-- Add full-text search on transcripts (PostgreSQL)
```

```typescript
// Track transcripts in memory during session
const sessionTranscripts = new Map<string, Array<{
  speaker: 'tutor' | 'student';
  text: string;
  timestamp: string;
  messageId: string;
}>>();

app.ws('/ws/realtime/:sessionId', (ws, req) => {
  const sessionId = req.params.sessionId;
  const userId = req.user?.id;
  
  // Initialize transcript array
  sessionTranscripts.set(sessionId, []);
  
  // Listen for OpenAI messages (from data channel or websocket)
  ws.on('message', (data) => {
    try {
      const message = JSON.parse(data.toString());
      const transcript = sessionTranscripts.get(sessionId) || [];
      
      // Capture tutor responses
      if (message.type === 'response.audio_transcript.done' || 
          message.type === 'conversation.item.created') {
        
        const tutorText = message.item?.content?.find(c => c.type === 'text')?.text ||
                         message.transcript;
        
        if (tutorText) {
          transcript.push({
            speaker: 'tutor',
            text: tutorText,
            timestamp: new Date().toISOString(),
            messageId: message.item?.id || message.id
          });
          
          console.log('💬 Tutor:', tutorText.substring(0, 50) + '...');
        }
      }
      
      // Capture student input (from speech-to-text)
      if (message.type === 'conversation.item.input_audio_transcription.completed' ||
          message.type === 'input_audio_buffer.speech_stopped') {
        
        const studentText = message.transcript;
        
        if (studentText) {
          transcript.push({
            speaker: 'student',
            text: studentText,
            timestamp: new Date().toISOString(),
            messageId: message.item_id || message.id
          });
          
          console.log('🎤 Student:', studentText.substring(0, 50) + '...');
        }
      }
      
      sessionTranscripts.set(sessionId, transcript);
      
    } catch (error) {
      console.error('Error processing transcript:', error);
    }
  });
  
  // Save transcript when session ends
  ws.on('close', async () => {
    const transcript = sessionTranscripts.get(sessionId) || [];
    const session = activeSessions.get(sessionId);
    
    if (!session) return;
    
    try {
      console.log('💾 Saving transcript with', transcript.length, 'messages');
      
      // Generate AI summary (optional but recommended)
      const summary = await generateSessionSummary(transcript);
      
      // Save to database
      await db.query(`
        UPDATE realtime_sessions 
        SET 
          transcript = $1,
          summary = $2,
          total_messages = $3,
          status = 'completed'
        WHERE id = $4
      `, [
        JSON.stringify(transcript),
        summary,
        transcript.length,
        sessionId
      ]);
      
      console.log('✅ Transcript saved successfully');
      
    } catch (error) {
      console.error('❌ Failed to save transcript:', error);
    } finally {
      // Cleanup memory
      sessionTranscripts.delete(sessionId);
      activeSessions.delete(sessionId);
    }
  });
});

// Helper: Generate AI summary of the session
async function generateSessionSummary(
  transcript: Array<{ speaker: string; text: string }>
): Promise<string> {
  if (transcript.length === 0) return 'No conversation recorded';
  
  // Combine all messages
  const fullConversation = transcript
    .map(t => `${t.speaker}: ${t.text}`)
    .join('\n');
  
  // Call OpenAI to generate summary (optional)
  try {
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        model: 'gpt-4o-mini',
        messages: [{
          role: 'system',
          content: 'Summarize this tutoring session in 2-3 sentences. Focus on topics covered and student progress.'
        }, {
          role: 'user',
          content: fullConversation.substring(0, 3000) // Limit tokens
        }],
        max_tokens: 150
      })
    });
    
    const data = await response.json();
    return data.choices[0].message.content;
    
  } catch (error) {
    console.error('Failed to generate summary:', error);
    
    // Fallback: Extract topics mentioned
    const topics = new Set<string>();
    transcript.forEach(t => {
      if (t.text.toLowerCase().includes('math')) topics.add('Math');
      if (t.text.toLowerCase().includes('science')) topics.add('Science');
      if (t.text.toLowerCase().includes('reading')) topics.add('Reading');
      // Add more topic detection as needed
    });
    
    return topics.size > 0 
      ? `Discussed: ${Array.from(topics).join(', ')}`
      : `${transcript.length} messages exchanged`;
  }
}
```

```typescript
  }

// GET /api/sessions - All sessions (last 30 days) with filters
router.get('/', authenticateUser, async (req, res) => {
  try {
    const userId = req.user.id;
    const { 
      page = 1, 
      limit = 20,
      studentId,
      subject,
      search,
      startDate,
      endDate
    } = req.query;
    
    const offset = (Number(page) - 1) * Number(limit);
    
    // Build dynamic query
    let query = `
      SELECT 
        id,
        student_id,
        student_name,
        subject,
        language,
        age_group,
        started_at,
        ended_at,
        minutes_used,
        summary,
        total_messages,
        status
      FROM realtime_sessions
      WHERE user_id = $1
        AND status = 'completed'
        AND started_at > NOW() - INTERVAL '30 days'
    `;
    
    const params: any[] = [userId];
    let paramCount = 1;
    
    if (studentId) {
      paramCount++;
      query += ` AND student_id = $${paramCount}`;
      params.push(studentId);
    }
    
    if (subject) {
      paramCount++;
      query += ` AND subject = $${paramCount}`;
      params.push(subject);
    }
    
    if (search) {
      paramCount++;
      query += ` AND (
        summary ILIKE $${paramCount} OR 
        to_tsvector('english', transcript::text) @@ plainto_tsquery('english', $${paramCount})
      )`;
      params.push(`%${search}%`);
    }
    
    if (startDate) {
      paramCount++;
      query += ` AND started_at >= $${paramCount}`;
      params.push(startDate);
    }
    
    if (endDate) {
      paramCount++;
      query += ` AND started_at <= $${paramCount}`;
      params.push(endDate);
    }
    
    // Get total count
    const countResult = await db.query(
      query.replace('SELECT id,', 'SELECT COUNT(*) as total FROM (SELECT id,') + `) as count_query`,
      params
    );
    const totalCount = parseInt(countResult.rows[0]?.total || '0');
    
    // Get paginated results
    query += ` ORDER BY started_at DESC LIMIT $${paramCount + 1} OFFSET $${paramCount + 2}`;
    params.push(Number(limit), offset);
    
    const result = await db.query(query, params);
    
    res.json({
      sessions: result.rows,
      pagination: {
        page: Number(page),
        limit: Number(limit),
        total: totalCount,
        totalPages: Math.ceil(totalCount / Number(limit))
      }
    });
    
  } catch (error) {
    console.error('Failed to fetch sessions:', error);
    res.status(500).json({ error: 'Failed to fetch sessions' });
  }
});

// GET /api/sessions/:id - Get full session with transcript
router.get('/:id', authenticateUser, async (req, res) => {
  try {
    const { id } = req.params;
    const userId = req.user.id;
    
    const result = await db.query(`
      SELECT 
        id,
        user_id,
        student_id,
        student_name,
        subject,
        language,
        age_group,
        voice,
        model,
        started_at,
        ended_at,
        minutes_used,
        transcript,
        summary,
        total_messages,
        audio_url,
        status,
        created_at
      FROM realtime_sessions
      WHERE id = $1 AND user_id = $2
    `, [id, userId]);
    
    if (result.rows.length === 0) {
      return res.status(404).json({ error: 'Session not found' });
    }
    
    res.json({ session: result.rows[0] });
    
  } catch (error) {
    console.error('Failed to fetch session:', error);
    res.status(500).json({ error: 'Failed to fetch session' });
  }
});

// DELETE /api/sessions/:id - Delete a session
router.delete('/:id', authenticateUser, async (req, res) => {
  try {
    const { id } = req.params;
    const userId = req.user.id;
    
    const result = await db.query(`
      DELETE FROM realtime_sessions
      WHERE id = $1 AND user_id = $2
      RETURNING id
    `, [id, userId]);
    
    if (result.rows.length === 0) {
      return res.status(404).json({ error: 'Session not found' });
    }
    
    res.json({ success: true, deletedId: id });
    
  } catch (error) {
    console.error('Failed to delete session:', error);
    res.status(500).json({ error: 'Failed to delete session' });
  }
});
```

### Create Transcript Viewer

```typescript
import { useEffect, useState } from 'react';
import { useParams, useNavigate } from 'react-router-dom';

interface TranscriptMessage {
  speaker: 'tutor' | 'student';
  text: string;
  timestamp: string;
  messageId: string;
}

interface SessionDetail {
  id: string;
  student_name: string;
  subject: string;
  language: string;
  started_at: string;
  ended_at: string;
  minutes_used: number;
  transcript: TranscriptMessage[];
  summary: string;
  total_messages: number;
}

export default function SessionDetail() {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();
  const [session, setSession] = useState<SessionDetail | null>(null);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    fetchSession();
  }, [id]);
  
  const fetchSession = async () => {
    try {
      const response = await fetch(`/api/sessions/${id}`);
      const data = await response.json();
      setSession(data.session);
    } catch (error) {
      console.error('Failed to fetch session:', error);
    } finally {
      setLoading(false);
    }
  };
  
  const formatTime = (timestamp: string) => {
    const date = new Date(timestamp);
    return date.toLocaleTimeString('en-US', { 
      hour: '2-digit', 
      minute: '2-digit',
      second: '2-digit'
    });
  };
  
  const exportTranscript = () => {
    if (!session) return;
    
    const text = session.transcript
      .map(m => `[${formatTime(m.timestamp)}] ${m.speaker === 'tutor' ? 'Tutor' : 'Student'}: ${m.text}`)
      .join('\n');
    
    const blob = new Blob([text], { type: 'text/plain' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `session-${session.id}-transcript.txt`;
    a.click();
  };
  
  if (loading) return <div className="loading">Loading session...</div>;
  if (!session) return <div className="error">Session not found</div>;
  
  return (
    <div className="session-detail">
      {/* Header */}
      <div className="detail-header">
        <button onClick={() => navigate('/sessions')} className="back-btn">
          ← Back to Sessions
        </button>
        
        <div className="session-meta">
          <h1>{session.subject} Session</h1>
          <p>
            {session.student_name || 'Student'} • 
            {new Date(session.started_at).toLocaleDateString()} • 
            {session.minutes_used} minutes • 
            {session.total_messages} messages
          </p>
        </div>
        
        <button onClick={exportTranscript} className="export-btn">
          Export Transcript
        </button>
      </div>
      
      {/* Summary */}
      {session.summary && (
        <div className="summary-box">
          <h3>Session Summary</h3>
          <p>{session.summary}</p>
        </div>
      )}
      
      {/* Transcript */}
      <div className="transcript-container">
        <h3>Full Transcript</h3>
        <div className="transcript">
          {session.transcript.map((message, index) => (
            <div 
              key={message.messageId || index} 
              className={`message ${message.speaker}`}
            >
              <div className="message-header">
                <span className="speaker">
                  {message.speaker === 'tutor' ? '🤖 Tutor' : '👤 Student'}
                </span>
                <span className="time">{formatTime(message.timestamp)}</span>
              </div>
              <div className="message-text">{message.text}</div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}
```

## Part 5: Auto-Cleanup (Delete sessions > 30 days)

**Create file: `server/src/jobs/cleanup-old-sessions.ts`**

```typescript
import { db } from '../db';

export async function cleanupOldSessions() {
  try {
    console.log('🗑️ Running session cleanup job...');
    
    const result = await db.query(`
      DELETE FROM realtime_sessions
      WHERE started_at < NOW() - INTERVAL '30 days'
        AND status = 'completed'
      RETURNING id
    `);
    
    const deletedCount = result.rows.length;
    
    if (deletedCount > 0) {
      console.log(`✅ Deleted ${deletedCount} sessions older than 30 days`);
    } else {
      console.log('✅ No old sessions to delete');
    }
    
    return deletedCount;
    
  } catch (error) {
    console.error('❌ Failed to cleanup old sessions:', error);
    throw error;
  }
}

// Run cleanup daily
if (process.env.NODE_ENV === 'production') {
  setInterval(
    cleanupOldSessions,
    24 * 60 * 60 * 1000 // 24 hours
  );
  
  // Run on startup
  cleanupOldSessions();
}
```
