## Fix #4: Verify API Endpoint Exists

### Check: `server/routes.ts`

Make sure this endpoint is defined:

```typescript
// POST /api/session/realtime/start - Start realtime voice session
app.post("/api/session/realtime/start", requireAuth, async (req, res) => {
  try {
    const { studentId, subject, gradeLevel, language } = req.body;
    const userId = req.session.userId;
    
    if (!userId) {
      return res.status(401).json({ error: "Not authenticated" });
    }
    
    // Validate user has credits
    const user = await db.query.users.findFirst({
      where: eq(users.id, userId)
    });
    
    if (!user || user.credits <= 0) {
      return res.status(403).json({ 
        error: "No voice minutes remaining. Please purchase more minutes to continue." 
      });
    }
    
    // Create session record
    const session = await db.insert(realtimeSessions).values({
      userId: userId,
      studentId: studentId || null,
      subject: subject || 'General',
      gradeLevel: gradeLevel || 'General',
      language: language || 'en',
      status: 'active',
      transcript: [],
      contentDocuments: [],
      startedAt: new Date()
    }).returning();
    
    // Return session details for frontend
    res.json({
      sessionId: session[0].id,
      message: "Session created successfully",
      creditsRemaining: user.credits
    });
    
  } catch (error) {
    console.error('Error starting realtime session:', error);
    res.status(500).json({ 
      error: "Failed to start realtime session",
      details: error.message 
    });
  }
});
```

app.get("/api/documents", requireAuth, async (req, res) => {
  try {
    const userId = req.session.userId;
    
    const userDocuments = await db.query.documents.findMany({
      where: eq(documents.userId, userId!),
      orderBy: desc(documents.uploadedAt)
    });
    
    res.json({ documents: userDocuments });
    
  } catch (error) {
