        purchased_minutes_balance, email_verified,

    const user = result.rows[0];

    // Calculate available minutes

### 3.3: Check Session Availability Endpoint

File: `server/routes/session.js` (or create if doesn't exist)

```javascript
const express = require('express');
const router = express.Router();

// GET /api/session/check-availability - Check if user can start a session
router.get('/check-availability', async (req, res) => {
  try {
    if (!req.session?.userId) {
      return res.status(401).json({ 
        available: false, 
        error: 'Not authenticated' 
      });
    }

    // Get user's current usage
    const result = await pool.query(
      `SELECT 
        subscription_minutes_used,
        subscription_minutes_limit,
        purchased_minutes_balance,
        subscription_status
      FROM users 
      WHERE id = $1`,
      [req.session.userId]
    );

    if (result.rows.length === 0) {
      return res.status(404).json({ 
        available: false, 
        error: 'User not found' 
      });
    }

    const user = result.rows[0];

    // Calculate available minutes
    const subscriptionMinutesLeft = user.subscription_minutes_limit - user.subscription_minutes_used;
    const totalAvailableMinutes = subscriptionMinutesLeft + user.purchased_minutes_balance;

    // Check if user has active subscription or purchased minutes
    const hasActiveSubscription = user.subscription_status === 'active';
    const hasPurchasedMinutes = user.purchased_minutes_balance > 0;
    const hasMinutesLeft = totalAvailableMinutes > 0;

    const available = (hasActiveSubscription || hasPurchasedMinutes) && hasMinutesLeft;

    res.json({
      available: available,
      minutesAvailable: Math.max(0, totalAvailableMinutes),
      subscriptionStatus: user.subscription_status,
      message: available 
        ? 'You can start a tutoring session' 
        : 'No minutes available. Please purchase more or upgrade your plan.'
    });

  } catch (error) {
    console.error('Error checking session availability:', error);
    res.status(500).json({ 
      available: false, 
      error: 'Failed to check availability' 
    });
  }
});

module.exports = router;
```
