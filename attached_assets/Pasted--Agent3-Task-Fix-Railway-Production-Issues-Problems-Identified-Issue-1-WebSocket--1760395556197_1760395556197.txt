## Fix #1: Stop WebSocket Loop & Enable Greeting

### Problem
Railway logs show continuous WebSocket connect/close cycles. The greeting never plays because:
- Connection drops before session establishes
- Path mismatch between client and server
- No heartbeat keeping connection alive
- Missing OpenAI Beta header

### Solution Steps

#### Step 1: Add WebSocket Logging

**File: `server/src/routes/realtime.ts` (or wherever WS handler lives)**

```typescript
// Add detailed logging to diagnose connection issues
app.ws('/ws/realtime/:sessionId', (ws, req) => {
  const sessionId = req.params.sessionId;
  const token = req.query.token;
  
  console.log('🔵 [WS] New connection attempt', { sessionId, hasToken: !!token });

  // Log closure with reason
  ws.on('close', (code, reason) => {
    console.warn('⚠️ [WS] Connection closed', { 
      sessionId, 
      code, 
      reason: reason?.toString() || 'no reason provided'
    });
  });

  ws.on('error', (err) => {
    console.error('❌ [WS] Error', { sessionId, error: err.message });
  });

  // Validate before proceeding
  if (!process.env.REALTIME_ENABLED || process.env.REALTIME_ENABLED !== 'true') {
    console.warn('⚠️ [WS] Rejected: REALTIME_DISABLED');
    ws.close(1008, 'Feature disabled');
    return;
  }

  if (!token || !isValidToken(token)) {
    console.warn('⚠️ [WS] Rejected: Invalid token');
    ws.close(1008, 'Invalid authentication');
    return;
  }

  console.log('✅ [WS] Connection accepted', { sessionId });
  
  // ... rest of your handler
});
```

#### Step 2: Add WebSocket Heartbeat (Prevent Idle Disconnects)

**In the same WebSocket handler:**

```typescript
// Add heartbeat to keep connection alive
const HEARTBEAT_INTERVAL = 15000; // 15 seconds

ws.isAlive = true;

ws.on('pong', () => {
  ws.isAlive = true;
});

const heartbeatInterval = setInterval(() => {
  if (ws.readyState !== ws.OPEN) {
    clearInterval(heartbeatInterval);
    return;
  }

  if (!ws.isAlive) {
    console.warn('⚠️ [WS] No pong received, terminating', { sessionId });
    ws.terminate();
    clearInterval(heartbeatInterval);
    return;
  }

  ws.isAlive = false;
  ws.ping();
}, HEARTBEAT_INTERVAL);

ws.on('close', () => {
  clearInterval(heartbeatInterval);
});
```

#### Step 2: Create Documents Upload Route

**Create file: `server/src/routes/documents.ts`**

```typescript
import { Router } from 'express';
import multer from 'multer';
import { v4 as uuidv4 } from 'uuid';

const router = Router();

// Configure multer for memory storage
const upload = multer({
  storage: multer.memoryStorage(),
  limits: {
    fileSize: 20 * 1024 * 1024, // 20MB max
  },
  fileFilter: (req, file, cb) => {
    // Accept common document types
    const allowedMimes = [
      'application/pdf',
      'application/msword',
      'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
      'text/plain',
      'text/csv',
      'image/jpeg',
      'image/png',
    ];
    
    if (allowedMimes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('Invalid file type'));
    }
  }
});

// POST /api/documents - Upload a document
router.post('/documents', upload.single('file'), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: 'No file provided' });
    }

    console.log('📄 File upload received:', {
      name: req.file.originalname,
      size: req.file.size,
      mime: req.file.mimetype
    });

    // TODO: Save to database and/or S3
    // For now, we'll save metadata to DB
    const doc = {
      id: uuidv4(),
      userId: req.user?.id, // if you have auth middleware
      name: req.file.originalname,
      mimeType: req.file.mimetype,
      size: req.file.size,
      uploadedAt: new Date(),
      // buffer: req.file.buffer // Store in S3 or disk instead
    };

    // If you have a documents table:
    // await db.documents.create({ data: doc });

    console.log('✅ Document saved:', doc.id);

    res.json({
      id: doc.id,
      name: doc.name,
      size: doc.size,
      uploadedAt: doc.uploadedAt
    });

  } catch (error) {
    console.error('❌ Document upload failed:', error);
    res.status(500).json({ 
      error: 'Upload failed',
      message: error.message 
    });
  }
});

// GET /api/documents - List user's documents
router.get('/documents', async (req, res) => {
  try {
    // TODO: Fetch from database

    console.error('❌ Failed to fetch documents:', error);
    res.status(500).json({ error: 'Failed to fetch documents' });
  }
});

router.delete('/documents/:id', async (req, res) => {

  }
});

export default router;
```

## Fix #3: Database Fail-Safe

### Prevent 500 Errors from Missing Tables

**File: `server/src/routes/realtime.ts`**

```typescript
async function saveRealtimeSession(sessionData: any) {
  try {
    await db.query(`
      INSERT INTO realtime_sessions (
        user_id, student_id, subject, language, 
        age_group, voice, model, status
      ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
    `, [
      sessionData.userId,
      sessionData.studentId,
      sessionData.subject,
      sessionData.language,
      sessionData.ageGroup,
      sessionData.voice,
      sessionData.model,
      'active'
    ]);
  } catch (error: any) {
    // PostgreSQL error code 42P01 = "undefined_table"
    if (error.code === '42P01') {
      console.warn('⚠️ realtime_sessions table missing; skipping DB save. Session continues normally.');
      return; // Don't throw - let voice session work
    }
    
    // Log but don't crash for other DB errors
    console.error('❌ Failed to save session to DB:', error.message);
    // Still don't throw - voice functionality is more important
  }
}
```
