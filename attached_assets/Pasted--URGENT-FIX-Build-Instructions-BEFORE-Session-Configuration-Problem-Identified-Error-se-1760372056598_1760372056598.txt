# URGENT FIX: Build Instructions BEFORE Session Configuration

## Problem Identified

**Error sequence from logs:**
```
1. "Configuring OpenAI with voice: echo" ⬅️ Sent config to OpenAI
2. OpenAI returns server_error ⬅️ No instructions provided yet!
3. "Built compact instructions for Robbie" ⬅️ TOO LATE!
```

**Root Cause:** Instructions are being built asynchronously AFTER the session configuration is sent to OpenAI, resulting in an empty or invalid `instructions` field.

---

## Fix: Ensure Synchronous Instruction Building

### Location: Wherever session configuration is happening (likely `server/routes.ts` or client-side WebRTC setup)

### WRONG (Current Code):

```typescript
// ❌ BAD - Sends config first, builds instructions later
async function startRealtimeSession() {
  // Configure session WITHOUT instructions ready
  const config = {
    voice: 'echo',
    instructions: buildInstructions(studentName) // Async, not awaited!
  };
  
  sendConfigToOpenAI(config); // Sent immediately
  
  // Instructions built AFTER config sent
  console.log('Built compact instructions for', studentName);
}
```

### CORRECT (Fixed Code):

```typescript
// ✅ GOOD - Builds instructions FIRST, then sends config
async function startRealtimeSession() {
  // 1. Build instructions FIRST (synchronously)
  const instructions = await buildInstructions(studentName, gradeLevel, subject);
  console.log('✅ Built instructions BEFORE config:', instructions.substring(0, 100));
  
  // 2. THEN configure session with complete instructions
  const config = {
    voice: 'echo',
    instructions: instructions // Ready to use!
  };
  
  // 3. Send complete config to OpenAI
  await sendConfigToOpenAI(config);
  console.log('✅ Configuration sent to OpenAI');
}
```

// SYNCHRONOUS instruction builder (not async!)
function buildSystemInstructions(
  studentName: string, 
  gradeLevel: string, 
  subject: string
): string {
  // Build instructions synchronously - no await, no promises
  const instructions = `You are an expert ${gradeLevel} tutor for ${subject}, speaking with ${studentName}.

PERSONALITY:
- Warm, encouraging, and patient
- Age-appropriate language for ${gradeLevel}
- Use the Socratic method to guide learning
- Celebrate effort and progress

TEACHING APPROACH:
1. Break complex topics into smaller steps
2. Use real-world examples appropriate for ${gradeLevel}
3. Check understanding frequently
4. Offer hints before giving answers
5. Keep responses concise (30-60 seconds per turn)

CONVERSATION STYLE:
- Natural and conversational
- Enthusiastic and supportive
- Adapt difficulty based on responses
- Ask follow-up questions

Your goal is to help ${studentName} understand concepts deeply, not just get the right answer.`;

  return instructions;
}

// Select voice based on grade level
function getVoiceForGrade(gradeLevel: string): string {
  const voiceMap: Record<string, string> = {
    'K-2': 'alloy',      // Clear, friendly
    '3-5': 'echo',       // Upbeat, engaging
    '6-8': 'fable',      // Warm, expressive
    '9-12': 'onyx',      // Deep, authoritative
    'College': 'nova',   // Professional, bright
    'General': 'echo'    // Default
  };
  
  return voiceMap[gradeLevel] || 'echo';
}

const startVoiceSession = async () => {
  try {
    setIsConnecting(true);
    
    console.log('🔄 Starting voice session...');
    
    // 1. Request session from backend (backend builds instructions first)
    const response = await fetch('/api/session/realtime/start', {
      method: 'POST',
      credentials: 'include',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        studentId: selectedStudent?.id,
        subject: selectedSubject,
        gradeLevel: selectedGrade
      })
    });
    
    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || 'Failed to start session');
    }
    
    const { clientSecret, voice, instructions } = await response.json();
    
    console.log('✅ Session created with voice:', voice);
    console.log('✅ Instructions length:', instructions?.length || 0);
    
    // 2. Set up WebRTC (instructions already configured on backend)
    await connectWebRTC(clientSecret);
    
    setIsConnecting(false);
    setIsSessionActive(true);
    
  } catch (error) {
    console.error('❌ Session start failed:', error);
    alert(`Failed to start session: ${error.message}`);
    setIsConnecting(false);
  }
};

const connectWebRTC = async (clientSecret: string) => {
  // WebRTC setup (instructions already sent to OpenAI by backend)
  const pc = new RTCPeerConnection();
  
  // Get microphone
  const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
  stream.getTracks().forEach(track => pc.addTrack(track, stream));
  
  // Create data channel
  const dataChannel = pc.createDataChannel('oai-events');
  
  dataChannel.onmessage = (event) => {
    const message = JSON.parse(event.data);
    handleRealtimeEvent(message);
  };
  
  // Create offer
  const offer = await pc.createOffer();
  await pc.setLocalDescription(offer);
  
  // Connect to OpenAI
  const sdpResponse = await fetch(
    'https://api.openai.com/v1/realtime?model=gpt-4o-realtime-preview-2024-10-01',
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${clientSecret}`,
        'Content-Type': 'application/sdp'
      },
      body: offer.sdp
    }
  );
  
  if (!sdpResponse.ok) {
    throw new Error('WebRTC connection failed');
  }
  
  const answerSDP = await sdpResponse.text();
  await pc.setRemoteDescription({ type: 'answer', sdp: answerSDP });
  
  // Handle incoming audio
  pc.ontrack = (event) => {
    if (audioRef.current) {
      audioRef.current.srcObject = event.streams[0];
      audioRef.current.play();
    }
  };
  
  console.log('✅ WebRTC connected');
};
